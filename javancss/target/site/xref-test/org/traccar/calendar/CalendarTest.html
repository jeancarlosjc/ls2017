14  import net.fortuna.ical4j.data.ParserException;
15  
16  public class CalendarTest {
17      
18      @Test
19      public void testCalendar() throws IOException, ParserException, ParseException, SQLException {
20          String calendarString = "BEGIN:VCALENDAR\n" + 
21                  "PRODID:-//Mozilla.org/NONSGML Mozilla Calendar V1.1//EN\n" + 
22                  "VERSION:2.0\n" + 
23                  "BEGIN:VTIMEZONE\n" + 
24                  "TZID:Asia/Yekaterinburg\n" + 
25                  "BEGIN:STANDARD\n" + 
26                  "TZOFFSETFROM:+0500\n" + 
27                  "TZOFFSETTO:+0500\n" + 
28                  "TZNAME:YEKT\n" + 
29                  "DTSTART:19700101T000000\n" + 
30                  "END:STANDARD\n" + 
31                  "END:VTIMEZONE\n" + 
32                  "BEGIN:VEVENT\n" + 
33                  "CREATED:20161213T045151Z\n" + 
34                  "LAST-MODIFIED:20161213T045242Z\n" + 
35                  "DTSTAMP:20161213T045242Z\n" + 
36                  "UID:9d000df0-6354-479d-a407-218dac62c7c9\n" + 
37                  "SUMMARY:Every night\n" + 
38                  "RRULE:FREQ=DAILY\n" + 
39                  "DTSTART;TZID=Asia/Yekaterinburg:20161130T230000\n" + 
40                  "DTEND;TZID=Asia/Yekaterinburg:20161201T070000\n" + 
41                  "TRANSP:OPAQUE\n" + 
42                  "END:VEVENT\n" + 
43                  "END:VCALENDAR";
44          Calendar calendar = new Calendar();
45          calendar.setData(calendarString.getBytes());
46          DateFormat format = new SimpleDateFormat("yyyy-MM-dd hh:mm:ssX");
47  
48          Date date = format.parse("2016-12-13 22:59:59+05");
49          Assert.assertTrue(!calendar.checkMoment(date));
50          date = format.parse("2016-12-13 23:00:01+05");
51          Assert.assertTrue(calendar.checkMoment(date));
52  
53          date = format.parse("2016-12-13 06:59:59+05");
54          Assert.assertTrue(calendar.checkMoment(date));
55          date = format.parse("2016-12-13 07:00:01+05");
56          Assert.assertTrue(!calendar.checkMoment(date));
57      }
