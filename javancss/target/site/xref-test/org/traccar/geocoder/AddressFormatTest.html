21          test(address, "%h %r %t %d %s %c %p", "1A Queen St Auckland NZ");
22          test(address, "%h %r %t", "1A Queen St Auckland");
23          test(address, "%h %r, %t", "1A Queen St, Auckland");
24          test(address, "%h %r, %t %p", "1A Queen St, Auckland");
25          test(address, "%t %d %c", "Auckland NZ");
26          test(address, "%t, %d, %c", "Auckland, NZ");
27          test(address, "%d %c", "NZ");
