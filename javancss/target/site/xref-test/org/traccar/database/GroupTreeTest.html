29      @Test
30      public void testGetDescendants() {
31          Collection<Group> groups = new ArrayList<>();
32          groups.add(createGroup(1, "First", 0));
33          groups.add(createGroup(2, "Second", 1));
34          groups.add(createGroup(3, "Third", 2));
35          groups.add(createGroup(4, "Fourth", 2));
36          groups.add(createGroup(5, "Fifth", 4));
37  
38          Collection<Device> devices = new ArrayList<>();
39          devices.add(createDevice(1, "One", 3));
40          devices.add(createDevice(2, "Two", 5));
41          devices.add(createDevice(3, "One", 5));
42  
43          GroupTree groupTree = new GroupTree(groups, devices);
44  
45          Assert.assertEquals(4, groupTree.getGroups(1).size());
46          Assert.assertEquals(3, groupTree.getGroups(2).size());
47          Assert.assertEquals(0, groupTree.getGroups(3).size());
48          Assert.assertEquals(1, groupTree.getGroups(4).size());
49  
50          Assert.assertEquals(3, groupTree.getDevices(1).size());
51          Assert.assertEquals(1, groupTree.getDevices(3).size());
