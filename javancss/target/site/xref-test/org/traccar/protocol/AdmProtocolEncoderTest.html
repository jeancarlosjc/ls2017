17  package org.traccar.protocol;
18  
19  import org.junit.Assert;
20  import org.junit.Test;
21  import org.traccar.ProtocolTest;
22  import org.traccar.model.Command;
23  
24  public class AdmProtocolEncoderTest extends ProtocolTest {
25  
26      @Test
27      public void testEncode() throws Exception {
28  
29          AdmProtocolEncoder encoder = new AdmProtocolEncoder();
30  
31          Command command = new Command();
32          command.setDeviceId(1);
33          command.setType(Command.TYPE_GET_DEVICE_STATUS);
34          Assert.assertEquals("STATUS\r\n", encoder.encodeCommand(command));
35  
36          command = new Command();
37          command.setDeviceId(1);
38          command.setType(Command.TYPE_CUSTOM);
39          command.set(Command.KEY_DATA, "INPUT 0");
40          Assert.assertEquals("INPUT 0\r\n", encoder.encodeCommand(command));
41      }
42  
43  }
