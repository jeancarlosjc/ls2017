16  package org.traccar.protocol;
17  
18  import java.util.TimeZone;
19  
20  import org.traccar.StringProtocolEncoder;
21  import org.traccar.helper.Log;
22  import org.traccar.model.Command;
23  
24  public class Pt502ProtocolEncoder extends StringProtocolEncoder implements StringProtocolEncoder.ValueFormatter {
25  
26      @Override
27      public String formatValue(String key, Object value) {
28          if (key.equals(Command.KEY_TIMEZONE)) {
29              return String.valueOf(TimeZone.getTimeZone((String) value).getRawOffset() / 3600000);
30          }
31  
32          return null;
33      }
34  
35      @Override
36      protected String formatCommand(Command command, String format, String... keys) {
37          return formatCommand(command, format, this, keys);
38      }
39  
40      @Override
41      protected Object encodeCommand(Command command) {
42  
43          switch (command.getType()) {
44              case Command.TYPE_OUTPUT_CONTROL:
45                  return formatCommand(command, "#OPC{%s},{%s}\r\n", Command.KEY_INDEX, Command.KEY_DATA);
46              case Command.TYPE_SET_TIMEZONE:
47                  return formatCommand(command, "#TMZ{%s}\r\n", Command.KEY_TIMEZONE);
48              case Command.TYPE_ALARM_SPEED:
49                  return formatCommand(command, "#SPD{%s}\r\n", Command.KEY_DATA);
50              case Command.TYPE_REQUEST_PHOTO:
51                  return formatCommand(command, "#PHO\r\n");
52              default:
53                  Log.warning(new UnsupportedOperationException(command.getType()));
54                  break;
55          }
56  
57          return null;
58      }
59  
60  }
