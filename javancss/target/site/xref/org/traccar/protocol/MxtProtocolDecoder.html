16  package org.traccar.protocol;
17  
18  import org.jboss.netty.buffer.ChannelBuffer;
19  import org.jboss.netty.buffer.ChannelBuffers;
20  import org.jboss.netty.channel.Channel;
21  import org.traccar.BaseProtocolDecoder;
22  import org.traccar.DeviceSession;
23  import org.traccar.helper.BitUtil;
24  import org.traccar.helper.Checksum;
25  import org.traccar.helper.DateBuilder;
26  import org.traccar.helper.UnitsConverter;
27  import org.traccar.model.Position;
28  
29  import java.net.SocketAddress;
30  import java.nio.ByteOrder;
31  
32  public class MxtProtocolDecoder extends BaseProtocolDecoder {
33  
34      public MxtProtocolDecoder(MxtProtocol protocol) {
35          super(protocol);
36      }
37  
38      public static final int MSG_ACK = 0x02;
39      public static final int MSG_NACK = 0x03;
40      public static final int MSG_POSITION = 0x31;
41  
42      private static void sendResponse(Channel channel, int device, long id, int crc) {
43          if (channel != null) {
44              ChannelBuffer response = ChannelBuffers.dynamicBuffer(ByteOrder.LITTLE_ENDIAN, 0);
45              response.writeByte(device);
46              response.writeByte(MSG_ACK);
47              response.writeInt((int) id);
48              response.writeShort(crc);
49              response.writeShort(Checksum.crc16(
50                      Checksum.CRC16_XMODEM, response.toByteBuffer()));
51  
52              ChannelBuffer encoded = ChannelBuffers.dynamicBuffer();
53              encoded.writeByte(0x01); // header
54              while (response.readable()) {
55                  int b = response.readByte();
56                  if (b == 0x01 || b == 0x04 || b == 0x10 || b == 0x11 || b == 0x13) {
57                      encoded.writeByte(0x10);
58                      b += 0x20;
59                  }
60                  encoded.writeByte(b);
61              }
62              encoded.writeByte(0x04); // ending
63              channel.write(encoded);
64          }
65      }
66  
67      @Override
68      protected Object decode(
69              Channel channel, SocketAddress remoteAddress, Object msg) throws Exception {
70  
71          ChannelBuffer buf = (ChannelBuffer) msg;
72  
73          buf.readUnsignedByte(); // start
74          int device = buf.readUnsignedByte(); // device descriptor
75          int type = buf.readUnsignedByte();
76  
77          long id = buf.readUnsignedInt();
78          DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, String.valueOf(id));
79          if (deviceSession == null) {
80              return null;
81          }
82  
83          if (type == MSG_POSITION) {
84  
85              Position position = new Position();
86              position.setProtocol(getProtocolName());
87              position.setDeviceId(deviceSession.getDeviceId());
88  
89              buf.readUnsignedByte(); // protocol
90              int infoGroups = buf.readUnsignedByte();
91  
92              position.set(Position.KEY_INDEX, buf.readUnsignedShort());
93  
94              DateBuilder dateBuilder = new DateBuilder().setDate(2000, 1, 1);
95  
96              long date = buf.readUnsignedInt();
97  
98              long days = BitUtil.from(date, 6 + 6 + 5);
99              long hours = BitUtil.between(date, 6 + 6, 6 + 6 + 5);
100             long minutes = BitUtil.between(date, 6, 6 + 6);
101             long seconds = BitUtil.to(date, 6);
102 
103             dateBuilder.addMillis((((days * 24 + hours) * 60 + minutes) * 60 + seconds) * 1000);
104 
105             position.setTime(dateBuilder.getDate());
106 
107             position.setValid(true);
108             position.setLatitude(buf.readInt() / 1000000.0);
109             position.setLongitude(buf.readInt() / 1000000.0);
110 
111             long flags = buf.readUnsignedInt();
112             position.set(Position.KEY_IGNITION, BitUtil.check(flags, 0));
113             if (BitUtil.check(flags, 1)) {
114                 position.set(Position.KEY_ALARM, Position.ALARM_GENERAL);
115             }
116             position.set(Position.KEY_INPUT, BitUtil.between(flags, 2, 7));
117             position.set(Position.KEY_OUTPUT, BitUtil.between(flags, 7, 10));
118             position.setCourse(BitUtil.between(flags, 10, 13) * 45);
119             //position.setValid(BitUtil.check(flags, 15));
120             position.set(Position.KEY_CHARGE, BitUtil.check(flags, 20));
121 
122             position.setSpeed(UnitsConverter.knotsFromKph(buf.readUnsignedByte()));
123 
124             buf.readUnsignedByte(); // input mask
125 
126             if (BitUtil.check(infoGroups, 0)) {
127                 buf.skipBytes(8); // waypoints
128             }
129 
130             if (BitUtil.check(infoGroups, 1)) {
131                 buf.skipBytes(8); // wireless accessory
132             }
133 
134             if (BitUtil.check(infoGroups, 2)) {
135                 position.set(Position.KEY_SATELLITES, buf.readUnsignedByte());
136                 position.set(Position.KEY_HDOP, buf.readUnsignedByte());
137                 position.setAccuracy(buf.readUnsignedByte());
138                 position.set(Position.KEY_RSSI, buf.readUnsignedByte());
139                 buf.readUnsignedShort(); // time since boot
140                 position.set(Position.KEY_POWER, buf.readUnsignedByte());
141                 position.set(Position.PREFIX_TEMP + 1, buf.readByte());
142             }
143 
144             if (BitUtil.check(infoGroups, 3)) {
145                 position.set(Position.KEY_ODOMETER, buf.readUnsignedInt());
146             }
147 
148             if (BitUtil.check(infoGroups, 4)) {
149                 position.set(Position.KEY_HOURS, buf.readUnsignedInt());
150             }
151 
152             if (BitUtil.check(infoGroups, 5)) {
153                 buf.readUnsignedInt(); // reason
154             }
155 
156             if (BitUtil.check(infoGroups, 6)) {
157                 position.set(Position.KEY_POWER, buf.readUnsignedShort() * 0.001);
158                 position.set(Position.KEY_BATTERY, buf.readUnsignedShort());
159             }
160 
161             if (BitUtil.check(infoGroups, 7)) {
162                 position.set(Position.KEY_DRIVER_UNIQUE_ID, String.valueOf(buf.readUnsignedInt()));
163             }
164 
165             buf.readerIndex(buf.writerIndex() - 3);
166             sendResponse(channel, device, id, buf.readUnsignedShort());
167 
168             return position;
169         }
170 
171         return null;
172     }
173 
174 }
