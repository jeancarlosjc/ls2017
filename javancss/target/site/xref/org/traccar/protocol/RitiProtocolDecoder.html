16  package org.traccar.protocol;
17  
18  import org.jboss.netty.buffer.ChannelBuffer;
19  import org.jboss.netty.channel.Channel;
20  import org.traccar.BaseProtocolDecoder;
21  import org.traccar.DeviceSession;
22  import org.traccar.helper.DateBuilder;
23  import org.traccar.helper.Parser;
24  import org.traccar.helper.PatternBuilder;
25  import org.traccar.model.Position;
26  
27  import java.net.SocketAddress;
28  import java.nio.charset.StandardCharsets;
29  import java.util.regex.Pattern;
30  
31  public class RitiProtocolDecoder extends BaseProtocolDecoder {
32  
33      public RitiProtocolDecoder(RitiProtocol protocol) {
34          super(protocol);
35      }
36  
37      private static final Pattern PATTERN = new PatternBuilder()
38              .text("$GPRMC,")
39              .number("(dd)(dd)(dd).?d*,")         // time (hhmmss)
40              .expression("([AV]),")               // validity
41              .number("(dd)(dd.d+),")              // latitude
42              .expression("([NS]),")
43              .number("(ddd)(dd.d+),")             // longitude
44              .expression("([EW]),")
45              .number("(d+.?d*)?,")                // speed
46              .number("(d+.?d*)?,")                // course
47              .number("(dd)(dd)(dd)")              // date (ddmmyy)
48              .any()
49              .compile();
50  
51      @Override
52      protected Object decode(
53              Channel channel, SocketAddress remoteAddress, Object msg) throws Exception {
54  
55          ChannelBuffer buf = (ChannelBuffer) msg;
56  
57          buf.skipBytes(2); // header
58  
59          Position position = new Position();
60          position.setProtocol(getProtocolName());
61  
62          DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, String.valueOf(buf.readUnsignedShort()));
63          if (deviceSession == null) {
64              return null;
65          }
66          position.setDeviceId(deviceSession.getDeviceId());
67  
68          position.set("mode", buf.readUnsignedByte());
69          position.set(Position.KEY_COMMAND, buf.readUnsignedByte());
70          position.set(Position.KEY_POWER, buf.readUnsignedShort() * 0.001);
71  
72          buf.skipBytes(5);  // status
73          buf.readUnsignedShort();  // idleCount
74          buf.readUnsignedShort();  // idleTime in seconds
75  
76          position.set(Position.KEY_DISTANCE, buf.readUnsignedInt());
77          position.set(Position.KEY_ODOMETER_TRIP, buf.readUnsignedInt());
78  
79          // Parse GPRMC
80          int end = buf.indexOf(buf.readerIndex(), buf.writerIndex(), (byte) '*');
81          String gprmc = buf.toString(buf.readerIndex(), end - buf.readerIndex(), StandardCharsets.US_ASCII);
82          Parser parser = new Parser(PATTERN, gprmc);
83          if (!parser.matches()) {
84              return null;
85          }
86  
87          DateBuilder dateBuilder = new DateBuilder()
88                  .setTime(parser.nextInt(0), parser.nextInt(0), parser.nextInt(0));
89  
90          position.setValid(parser.next().equals("A"));
91          position.setLatitude(parser.nextCoordinate());
92          position.setLongitude(parser.nextCoordinate());
93          position.setSpeed(parser.nextDouble(0));
94          position.setCourse(parser.nextDouble(0));
95  
96          dateBuilder.setDateReverse(parser.nextInt(0), parser.nextInt(0), parser.nextInt(0));
97          position.setTime(dateBuilder.getDate());
98  
99          return position;
100     }
101 
102 }
