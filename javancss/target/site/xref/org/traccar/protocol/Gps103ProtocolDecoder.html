37      private static final Pattern PATTERN = new PatternBuilder()
38              .text("imei:")
39              .number("(d+),")                     // imei
40              .expression("([^,]+),")              // alarm
41              .number("(dd)/?(dd)/?(dd) ?")        // local date (yymmdd)
42              .number("(dd):?(dd)(?:dd)?,")        // local time (hhmmss)
43              .expression("([^,]+)?,")             // rfid
44              .expression("[FL],")                 // full / low
45              .groupBegin()
46              .number("(dd)(dd)(dd).d+")           // time utc (hhmmss)
47              .or()
48              .number("(?:d{1,5}.d+)?")
49              .groupEnd()
50              .text(",")
51              .expression("([AV]),")               // validity
52              .expression("([NS]),").optional()
53              .number("(d+)(dd.d+),")              // latitude (ddmm.mmmm)
54              .expression("([NS]),").optional()
55              .expression("([EW]),").optional()
56              .number("(d+)(dd.d+),")              // longitude (dddmm.mmmm)
57              .expression("([EW])?,").optional()
58              .number("(d+.?d*)?,?")               // speed
59              .number("(d+.?d*)?,?")               // course
60              .number("(d+.?d*)?,?")               // altitude
61              .expression("([^,;]+)?,?")
62              .expression("([^,;]+)?,?")
63              .expression("([^,;]+)?,?")
64              .expression("([^,;]+)?,?")
65              .expression("([^,;]+)?,?")
66              .any()
67              .compile();
68  
69      private static final Pattern PATTERN_NETWORK = new PatternBuilder()
70              .text("imei:")
71              .number("(d+),")                     // imei
72              .expression("[^,]+,")                // alarm
73              .number("d*,,")
74              .text("L,,,")
75              .number("(x+),,")                    // lac
76              .number("(x+),,,")                   // cid
77              .any()
78              .compile();
79  
80      private static final Pattern PATTERN_HANDSHAKE = new PatternBuilder()
81              .number("##,imei:(d+),A")
82              .compile();
83  
84      private static final Pattern PATTERN_OBD = new PatternBuilder()
85              .text("imei:")
86              .number("(d+),")                     // imei
87              .expression("OBD,")                  // type
88              .number("(dd)(dd)(dd)")              // date (yymmdd)
89              .number("(dd)(dd)(dd),")             // time (hhmmss)
90              .number("(d+),")                     // odometer
91              .number("(d+.d+)?,")                 // fuel instant
92              .number("(d+.d+)?,")                 // fuel average
93              .number("(d+)?,")                    // hours
94              .number("(d+),")                     // speed
95              .number("(d+.?d*%),")                // power load
96              .number("(?:([-+]?d+)|[-+]?),")      // temperature
97              .number("(d+.?d*%),")                // throttle
98              .number("(d+),")                     // rpm
99              .number("(d+.d+),")                  // battery
100             .number("([^;]*)")                   // dtcs
101             .any()
102             .compile();

104     private String decodeAlarm(String value) {
105         if (value.startsWith("T:")) {
106             return Position.ALARM_TEMPERATURE;
107         } else if (value.startsWith("oil")) {
108             return Position.ALARM_OIL_LEAK;
109         }
110         switch (value) {
111             case "tracker":
112                 return null;
113             case "help me":
114                 return Position.ALARM_SOS;
115             case "low battery":
116                 return Position.ALARM_LOW_BATTERY;
117             case "stockade":
118                 return Position.ALARM_GEOFENCE;
119             case "move":
120                 return Position.ALARM_MOVEMENT;
121             case "speed":
122                 return Position.ALARM_OVERSPEED;
123             case "acc on":
124                 return Position.ALARM_POWER_ON;
125             case "acc off":
126                 return Position.ALARM_POWER_OFF;
127             case "door alarm":
128                 return Position.ALARM_DOOR;
129             case "ac alarm":
130                 return Position.ALARM_POWER_CUT;
131             case "accident alarm":
132                 return Position.ALARM_ACCIDENT;
133             case "sensor alarm":
134                 return Position.ALARM_SHOCK;
135             case "bonnet alarm":
136                 return Position.ALARM_BONNET;
137             case "footbrake alarm":
138                 return Position.ALARM_FOOT_BRAKE;
139             case "DTC":
140                 return Position.ALARM_FAULT;
141             default:
142                 return null;
143         }
144     }

146     @Override
147     protected Object decode(
148             Channel channel, SocketAddress remoteAddress, Object msg) throws Exception {
149 
150         String sentence = (String) msg;
151 
152         // Send response #1
153         if (sentence.contains("##")) {
154             if (channel != null) {
155                 channel.write("LOAD", remoteAddress);
156                 Parser handshakeParser = new Parser(PATTERN_HANDSHAKE, sentence);
157                 if (handshakeParser.matches()) {
158                     getDeviceSession(channel, remoteAddress, handshakeParser.next());
159                 }
160             }
161             return null;
162         }
163 
164         // Send response #2
165         if (!sentence.isEmpty() && Character.isDigit(sentence.charAt(0))) {
166             if (channel != null) {
167                 channel.write("ON", remoteAddress);
168             }
169             int start = sentence.indexOf("imei:");
170             if (start >= 0) {
171                 sentence = sentence.substring(start);
172             } else {
173                 return null;
174             }
175         }
176 
177         Position position = new Position();
178         position.setProtocol(getProtocolName());
179 
180         Parser parser = new Parser(PATTERN_NETWORK, sentence);
181         if (parser.matches()) {
182 
183             DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, parser.next());
184             if (deviceSession == null) {
185                 return null;
186             }
187             position.setDeviceId(deviceSession.getDeviceId());
188 
189             getLastLocation(position, null);
190 
191             position.setNetwork(new Network(
192                     CellTower.fromLacCid(parser.nextHexInt(0), parser.nextHexInt(0))));
193 
194             return position;
195 
196         }
197 
198         parser = new Parser(PATTERN_OBD, sentence);
199         if (parser.matches()) {
200 
201             DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, parser.next());
202             if (deviceSession == null) {
203                 return null;
204             }
205             position.setDeviceId(deviceSession.getDeviceId());
206 
207             getLastLocation(position, parser.nextDateTime());
208 
209             position.set(Position.KEY_ODOMETER, parser.nextInt(0));
210             parser.nextDouble(0); // instant fuel consumption
211             position.set(Position.KEY_FUEL_CONSUMPTION, parser.nextDouble(0));
212             position.set(Position.KEY_HOURS, parser.nextInt());
213             position.set(Position.KEY_OBD_SPEED, parser.nextInt(0));
214             position.set(Position.KEY_ENGINE_LOAD, parser.next());
215             position.set(Position.KEY_COOLANT_TEMP, parser.nextInt());
216             position.set(Position.KEY_THROTTLE, parser.next());
217             position.set(Position.KEY_RPM, parser.nextInt(0));
218             position.set(Position.KEY_BATTERY, parser.nextDouble(0));
219             position.set(Position.KEY_DTCS, parser.next().replace(',', ' ').trim());
220 
221             return position;
222 
223         }
224 
225         parser = new Parser(PATTERN, sentence);
226         if (!parser.matches()) {
227             return null;
228         }
229 
230         String imei = parser.next();
231         DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, imei);
232         if (deviceSession == null) {
233             return null;
234         }
235         position.setDeviceId(deviceSession.getDeviceId());
236 
237         String alarm = parser.next();
238         position.set(Position.KEY_ALARM, decodeAlarm(alarm));
239         if (alarm.equals("help me")) {
240             if (channel != null) {
241                 channel.write("**,imei:" + imei + ",E;", remoteAddress);
242             }
243         } else if (alarm.equals("acc on")) {
244             position.set(Position.KEY_IGNITION, true);
245         } else if (alarm.equals("acc off")) {
246             position.set(Position.KEY_IGNITION, false);
247         } else if (alarm.startsWith("T:")) {
248             position.set(Position.PREFIX_TEMP + 1, alarm.substring(2));
249         } else if (alarm.startsWith("oil ")) {
250             position.set("oil", alarm.substring(4));
251         } else if (!position.getAttributes().containsKey(Position.KEY_ALARM) && !alarm.equals("tracker")) {
252             position.set(Position.KEY_EVENT, alarm);
253         }
254 
255         DateBuilder dateBuilder = new DateBuilder()
256                 .setDate(parser.nextInt(0), parser.nextInt(0), parser.nextInt(0));
257 
258         int localHours = parser.nextInt(0);
259         int localMinutes = parser.nextInt(0);
260 
261         String rfid = parser.next();
262         if (alarm.equals("rfid")) {
263             position.set(Position.KEY_DRIVER_UNIQUE_ID, rfid);
264         }
265 
266         String utcHours = parser.next();
267         String utcMinutes = parser.next();
268 
269         dateBuilder.setTime(localHours, localMinutes, parser.nextInt(0));
270 
271         // Timezone calculation
272         if (utcHours != null && utcMinutes != null) {
273             int deltaMinutes = (localHours - Integer.parseInt(utcHours)) * 60;
274             deltaMinutes += localMinutes - Integer.parseInt(utcMinutes);
275             if (deltaMinutes <= -12 * 60) {
276                 deltaMinutes += 24 * 60;
277             } else if (deltaMinutes > 12 * 60) {
278                 deltaMinutes -= 24 * 60;
279             }
280             dateBuilder.addMinute(-deltaMinutes);
281         }
282         position.setTime(dateBuilder.getDate());
283 
284         position.setValid(parser.next().equals("A"));
285         position.setLatitude(parser.nextCoordinate(Parser.CoordinateFormat.HEM_DEG_MIN_HEM));
286         position.setLongitude(parser.nextCoordinate(Parser.CoordinateFormat.HEM_DEG_MIN_HEM));
287         position.setSpeed(parser.nextDouble(0));
288         position.setCourse(parser.nextDouble(0));
289         position.setAltitude(parser.nextDouble(0));
290 
291         for (int i = 1; i <= 5; i++) {
292             position.set(Position.PREFIX_IO + i, parser.next());
293         }
294 
295         return position;
296     }
