38      private static final Pattern PATTERN1 = new PatternBuilder()
39              .text("$$")                          // header
40              .number("xx")                        // length
41              .number("(d+)|")                     // imei
42              .expression("(..)")                  // alarm
43              .text("$GPRMC,")
44              .number("(dd)(dd)(dd).d+,")          // time (hhmmss)
45              .expression("([AV]),")               // validity
46              .number("(d+)(dd.d+),([NS]),")       // latitude
47              .number("(d+)(dd.d+),([EW]),")       // longitude
48              .number("(d+.?d*)?,")                // speed
49              .number("(d+.?d*)?,")                // course
50              .number("(dd)(dd)(dd)")              // date (ddmmyy)
51              .expression("[^*]*").text("*")
52              .number("xx|")                       // checksum
53              .number("(d+.d+)|")                  // pdop
54              .number("(d+.d+)|")                  // hdop
55              .number("(d+.d+)|")                  // vdop
56              .number("(d+)|")                     // io status
57              .number("d+|")                       // battery time
58              .number("d")                         // charged
59              .number("(ddd)")                     // battery
60              .number("(dddd)|")                   // power
61              .number("(d+)|").optional()          // adc
62              .number("x*(xxxx)")                  // lac
63              .number("(xxxx)|")                   // cid
64              .number("(d+)|")                     // temperature
65              .number("(d+.d+)|")                  // odometer
66              .number("d+|")                       // serial number
67              .any()
68              .number("xxxx")                      // checksum
69              .any()
70              .compile();

73              .text("$$")                          // header
74              .number("xx")                        // length
75              .number("(d+)|")                     // imei
76              .expression("(..)")                  // alarm type
77              .number("(dd)(dd)(dd)")              // date (ddmmyy)
78              .number("(dd)(dd)(dd)|")             // time (hhmmss)
79              .expression("([AV])|")               // validity

82              .number("(d+)(dd.d+)|")              // longitude
83              .expression("([EW])|")

85              .number("(d+)?|")                    // course
86              .number("(d+.d+)|")                  // hdop
87              .number("(d+)|")                     // io status
88              .number("d")                         // charged
89              .number("(dd)")                      // battery
90              .number("(dd)|")                     // external power
91              .number("(d+)|")                     // adc
92              .number("(xxxx)")                    // lac
93              .number("(xxxx)|")                   // cid
94              .number("(d+)|")                     // temperature
95              .number("(d+.d+)|")                  // odometer
96              .number("d+|")                       // serial number
97              .number("xxxx")                      // checksum
98              .any()
99              .compile();

101     private static final Pattern PATTERN3 = new PatternBuilder()
102             .text("$$")                          // header
103             .number("xx")                        // length
104             .number("(d+)|")                     // imei
105             .expression("(..)")                  // alarm type
106             .number("(dd)(dd)(dd)")              // date (ddmmyy)
107             .number("(dd)(dd)(dd)")              // time (hhmmss)
108             .number("(xxxx)")                    // io status
109             .expression("[01]")                  // charging
110             .number("(dd)")                      // battery
111             .number("(dd)")                      // external power
112             .number("(dddd)")                    // adc 1
113             .number("(dddd)")                    // adc 2
114             .number("(ddd)")                     // temperature 1
115             .number("(ddd)")                     // temperature 2
116             .number("(xxxx)")                    // lac
117             .number("(xxxx)")                    // cid
118             .expression("([AV])")                // validity
119             .number("(dd)")                      // satellites
120             .number("(ddd)")                     // course
121             .number("(ddd)")                     // speed
122             .number("(dd.d)")                    // pdop
123             .number("(d{7})")                    // odometer
124             .number("(dd)(dd.dddd)([NS])")       // latitude
125             .number("(ddd)(dd.dddd)([EW])")      // longitude
126             .number("dddd")                      // serial number
127             .number("xxxx")                      // checksum
128             .any()
129             .compile();

131     private static final Pattern PATTERN4 = new PatternBuilder()
132             .text("$$")                          // header
133             .number("dddd")                      // length
134             .text("AA")                          // type
135             .number("(d+)|")                     // imei
136             .number("(x{8})")                    // status
137             .number("(dd)(dd)(dd)")              // date (yymmdd)
138             .number("(dd)(dd)(dd)")              // time (hhmmss)
139             .number("(dd)")                      // battery
140             .number("(dd)")                      // external power
141             .number("(dddd)")                    // adc 1
142             .groupBegin()
143             .groupBegin()
144             .number("(dddd)")                    // adc 2
145             .number("(dddd)")                    // adc 3
146             .number("(dddd)")                    // adc 4
147             .groupEnd("?")
148             .number("(dddd)")                    // temperature 1
149             .number("(dddd)")                    // temperature 2
150             .groupEnd("?")
151             .number("(xxxx)")                    // lac
152             .number("(xxxx)")                    // cid
153             .number("(dd)")                      // satellites
154             .number("(dd)")                      // gsm (rssi)
155             .number("(ddd)")                     // course
156             .number("(ddd)")                     // speed
157             .number("(dd.d)")                    // hdop
158             .number("(d{7})")                    // odometer
159             .number("(dd)(dd.dddd)([NS])")       // latitude
160             .number("(ddd)(dd.dddd)([EW])")      // longitude
161             .number("dddd")                      // serial number
162             .number("xx")                        // checksum
163             .any()
164             .compile();

166     private String decodeAlarm(Short value) {
167         switch (value) {
168             case 0x01:
169                 return Position.ALARM_SOS;
170             case 0x10:
171                 return Position.ALARM_LOW_BATTERY;
172             case 0x11:
173                 return Position.ALARM_OVERSPEED;
174             case 0x30:
175                 return Position.ALARM_PARKING;
176             case 0x42:
177                 return Position.ALARM_GEOFENCE_EXIT;
178             case 0x43:
179                 return Position.ALARM_GEOFENCE_ENTER;
180             default:
181                 return null;
182         }
183     }
184 
185     @Override
186     protected Object decode(
187             Channel channel, SocketAddress remoteAddress, Object msg) throws Exception {
188 
189         String sentence = (String) msg;
190 
191         // Determine format
192         Pattern pattern = PATTERN3;
193         if (sentence.indexOf("AA") == 6) {
194             pattern = PATTERN4;
195         } else if (sentence.contains("$GPRMC")) {
196             pattern = PATTERN1;
197         } else {
198             int index = sentence.indexOf('|');
199             if (index != -1 && sentence.indexOf('|', index + 1) != -1) {
200                 pattern = PATTERN2;
201             }
202         }
203 
204         Parser parser = new Parser(pattern, sentence);
205         if (!parser.matches()) {
206             return null;
207         }
208 
209         Position position = new Position();
210         position.setProtocol(getProtocolName());
211 
212         DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, parser.next());
213         if (deviceSession == null) {
214             return null;
215         }
216         position.setDeviceId(deviceSession.getDeviceId());
217 
218         if (pattern == PATTERN1 || pattern == PATTERN2) {
219             if (parser.hasNext()) {
220                 position.set(Position.KEY_ALARM, decodeAlarm(Short.parseShort(parser.next(), 16)));
221             }
222             DateBuilder dateBuilder = new DateBuilder();
223             int year = 0, month = 0, day = 0;
224             if (pattern == PATTERN2) {
225                 day   = parser.nextInt(0);
226                 month = parser.nextInt(0);
227                 year  = parser.nextInt(0);
228             }
229             dateBuilder.setTime(parser.nextInt(0), parser.nextInt(0), parser.nextInt(0));
230 
231             position.setValid(parser.next().equals("A"));
232             position.setLatitude(parser.nextCoordinate());
233             position.setLongitude(parser.nextCoordinate());
234             position.setSpeed(parser.nextDouble(0));
235             position.setCourse(parser.nextDouble(0));
236 
237             if (pattern == PATTERN1) {
238                 day   = parser.nextInt(0);
239                 month = parser.nextInt(0);
240                 year  = parser.nextInt(0);
241             }
242             if (year == 0) {
243                 return null; // ignore invalid data
244             }
245             dateBuilder.setDate(year, month, day);
246             position.setTime(dateBuilder.getDate());
247 
248             if (pattern == PATTERN1) {
249                 position.set(Position.KEY_PDOP, parser.nextDouble());
250                 position.set(Position.KEY_HDOP, parser.nextDouble());
251                 position.set(Position.KEY_VDOP, parser.nextDouble());
252             } else {
253                 position.set(Position.KEY_HDOP, parser.nextDouble());
254             }
255 
256             position.set(Position.PREFIX_IO + 1, parser.next());
257             if (pattern == PATTERN1) {
258                 position.set(Position.KEY_BATTERY, parser.nextDouble(0) * 0.01);
259             } else {
260                 position.set(Position.KEY_BATTERY, parser.nextDouble(0) * 0.1);
261             }
262             position.set(Position.KEY_POWER, parser.nextDouble(0));
263             position.set(Position.PREFIX_ADC + 1, parser.next());
264 
265             int lac = parser.nextHexInt(0);
266             int cid = parser.nextHexInt(0);
267             if (lac != 0 && cid != 0) {
268                 position.setNetwork(new Network(CellTower.fromLacCid(lac, cid)));
269             }
270 
271             position.set(Position.PREFIX_TEMP + 1, parser.next());
272             position.set(Position.KEY_ODOMETER, parser.nextDouble(0) * 1000);
273 
274         } else if (pattern == PATTERN3) {
275             if (parser.hasNext()) {
276                 position.set(Position.KEY_ALARM, decodeAlarm(Short.parseShort(parser.next(), 16)));
277             }
278 
279             position.setTime(parser.nextDateTime(Parser.DateTimeFormat.DMY_HMS));
280 
281             position.set(Position.PREFIX_IO + 1, parser.next());
282             position.set(Position.KEY_BATTERY, parser.nextDouble(0) * 0.1);
283             position.set(Position.KEY_POWER, parser.nextDouble(0));
284             position.set(Position.PREFIX_ADC + 1, parser.next());
285             position.set(Position.PREFIX_ADC + 2, parser.next());
286             position.set(Position.PREFIX_TEMP + 1, parser.next());
287             position.set(Position.PREFIX_TEMP + 2, parser.next());
288 
289             position.setNetwork(new Network(
290                     CellTower.fromLacCid(parser.nextHexInt(0), parser.nextHexInt(0))));
291 
292             position.setValid(parser.next().equals("A"));
293             position.set(Position.KEY_SATELLITES, parser.nextInt());
294             position.setCourse(parser.nextDouble(0));
295             position.setSpeed(parser.nextDouble(0));
296             position.set(Position.KEY_PDOP, parser.nextDouble());
297             position.set(Position.KEY_ODOMETER, parser.nextInt(0) * 1000);
298 
299             position.setLatitude(parser.nextCoordinate());
300             position.setLongitude(parser.nextCoordinate());
301 
302         } else if (pattern == PATTERN4) {
303             position.set(Position.KEY_STATUS, parser.next());
304 
305             position.setTime(parser.nextDateTime());
306 
307             position.set(Position.KEY_BATTERY, parser.nextDouble(0) * 0.1);
308             position.set(Position.KEY_POWER, parser.nextDouble(0));
309 
310             position.set(Position.PREFIX_ADC + 1, parser.next());
311             position.set(Position.PREFIX_ADC + 2, parser.next());
312             position.set(Position.PREFIX_ADC + 3, parser.next());
313             position.set(Position.PREFIX_ADC + 4, parser.next());
314             position.set(Position.PREFIX_TEMP + 1, parser.next());
315             position.set(Position.PREFIX_TEMP + 2, parser.next());
316 
317             CellTower cellTower = CellTower.fromLacCid(parser.nextHexInt(0), parser.nextHexInt(0));
318             position.set(Position.KEY_SATELLITES, parser.nextInt(0));
319             cellTower.setSignalStrength(parser.nextInt(0));
320             position.setNetwork(new Network(cellTower));
321 
322             position.setCourse(parser.nextDouble(0));
323             position.setSpeed(UnitsConverter.knotsFromKph(parser.nextDouble(0)));
324             position.set(Position.KEY_HDOP, parser.nextDouble(0));
325             position.set(Position.KEY_ODOMETER, parser.nextInt(0) * 1000);
326 
327             position.setValid(true);
328             position.setLatitude(parser.nextCoordinate());
329             position.setLongitude(parser.nextCoordinate());
330         }
331         if (channel != null) {
332             channel.write("ACK OK\r\n");
333         }
334         return position;
335     }
