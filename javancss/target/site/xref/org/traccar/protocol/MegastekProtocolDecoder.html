37      private static final Pattern PATTERN_GPRMC = new PatternBuilder()
38              .text("$GPRMC,")
39              .number("(dd)(dd)(dd).(ddd),")       // time (hhmmss.sss)
40              .expression("([AV]),")               // validity
41              .number("(d+)(dd.d+),([NS]),")       // latitude
42              .number("(d+)(dd.d+),([EW]),")       // longitude
43              .number("(d+.d+)?,")                 // speed
44              .number("(d+.d+)?,")                 // course
45              .number("(dd)(dd)(dd)")              // date (ddmmyy)
46              .any()                               // checksum
47              .compile();
48  
49      private static final Pattern PATTERN_SIMPLE = new PatternBuilder()
50              .expression("[FL],")                 // flag
51              .expression("([^,]*),")              // alarm
52              .number("imei:(d+),")                // imei
53              .number("(d+/?d*)?,")                // satellites
54              .number("(d+.d+)?,")                 // altitude
55              .number("Battery=(d+)%,,?")          // battery
56              .number("(d)?,")                     // charger
57              .number("(d+)?,")                    // mcc
58              .number("(d+)?,")                    // mnc
59              .number("(xxxx),")                   // lac
60              .number("(xxxx);")                   // cid
61              .any()                               // checksum
62              .compile();
63  
64      private static final Pattern PATTERN_ALTERNATIVE = new PatternBuilder()
65              .number("(d+),")                     // mcc
66              .number("(d+),")                     // mnc
67              .number("(xxxx),")                   // lac
68              .number("(xxxx),")                   // cid
69              .number("(d+),")                     // gsm signal
70              .number("(d+),")                     // battery
71              .number("(d+),")                     // flags
72              .number("(d+),")                     // inputs
73              .number("(?:(d+),)?")                // outputs
74              .number("(d.?d*),")                  // adc 1
75              .groupBegin()
76              .number("(d.dd),")                   // adc 2
77              .number("(d.dd),")                   // adc 3
78              .groupEnd("?")
79              .expression("([^;]+);")              // alarm
80              .any()                               // checksum
81              .compile();

83      private boolean parseLocation(String location, Position position) {
84  
85          Parser parser = new Parser(PATTERN_GPRMC, location);
86          if (!parser.matches()) {
87              return false;
88          }
89  
90          DateBuilder dateBuilder = new DateBuilder()
91                  .setTime(parser.nextInt(0), parser.nextInt(0), parser.nextInt(0), parser.nextInt(0));
92  
93          position.setValid(parser.next().equals("A"));
94          position.setLatitude(parser.nextCoordinate());
95          position.setLongitude(parser.nextCoordinate());
96          position.setSpeed(parser.nextDouble(0));
97          position.setCourse(parser.nextDouble(0));
98  
99          dateBuilder.setDateReverse(parser.nextInt(0), parser.nextInt(0), parser.nextInt(0));
100         position.setTime(dateBuilder.getDate());
101 
102         return true;
103     }

105     private Position decodeOld(Channel channel, SocketAddress remoteAddress, String sentence) {
106 
107         // Detect type
108         boolean simple = sentence.charAt(3) == ',' || sentence.charAt(6) == ',';
109 
110         // Split message
111         String id;
112         String location;
113         String status;
114         if (simple) {
115 
116             int beginIndex = sentence.indexOf(',') + 1;
117             int endIndex = sentence.indexOf(',', beginIndex);
118             id = sentence.substring(beginIndex, endIndex);
119 
120             beginIndex = endIndex + 1;
121             endIndex = sentence.indexOf('*', beginIndex);
122             if (endIndex != -1) {
123                 endIndex += 3;
124             } else {
125                 endIndex = sentence.length();
126             }
127             location = sentence.substring(beginIndex, endIndex);
128 
129             beginIndex = endIndex + 1;
130             if (beginIndex > sentence.length()) {
131                 beginIndex = endIndex;
132             }
133             status = sentence.substring(beginIndex);
134 
135         } else {
136 
137             int beginIndex = 3;
138             int endIndex = beginIndex + 16;
139             id = sentence.substring(beginIndex, endIndex).trim();
140 
141             beginIndex = endIndex + 2;
142             endIndex = sentence.indexOf('*', beginIndex) + 3;
143             location = sentence.substring(beginIndex, endIndex);
144 
145             beginIndex = endIndex + 1;
146             status = sentence.substring(beginIndex);
147 
148         }
149 
150         Position position = new Position();
151         position.setProtocol(getProtocolName());
152         if (!parseLocation(location, position)) {
153             return null;
154         }
155 
156         if (simple) {
157 
158             Parser parser = new Parser(PATTERN_SIMPLE, status);
159             if (parser.matches()) {
160 
161                 position.set(Position.KEY_ALARM, decodeAlarm(parser.next()));
162 
163                 DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, parser.next(), id);
164                 if (deviceSession == null) {
165                     return null;
166                 }
167                 position.setDeviceId(deviceSession.getDeviceId());
168 
169                 String sat = parser.next();
170                 if (sat.contains("/")) {
171                     position.set(Position.KEY_SATELLITES, Integer.parseInt(sat.split("/")[0]));
172                     position.set(Position.KEY_SATELLITES_VISIBLE, Integer.parseInt(sat.split("/")[1]));
173                 } else {
174                     position.set(Position.KEY_SATELLITES, Integer.parseInt(sat));
175                 }
176 
177                 position.setAltitude(parser.nextDouble(0));
178 
179                 position.set(Position.KEY_BATTERY_LEVEL, parser.nextDouble(0));
180 
181                 String charger = parser.next();
182                 if (charger != null) {
183                     position.set(Position.KEY_CHARGE, Integer.parseInt(charger) == 1);
184                 }
185 
186                 if (parser.hasNext(4)) {
187                     position.setNetwork(new Network(CellTower.from(
188                             parser.nextInt(0), parser.nextInt(0), parser.nextHexInt(0), parser.nextHexInt(0))));
189                 }
190 
191             } else {
192 
193                 DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, id);
194                 if (deviceSession == null) {
195                     return null;
196                 }
197                 position.setDeviceId(deviceSession.getDeviceId());
198 
199             }
200 
201         } else {
202 
203             Parser parser = new Parser(PATTERN_ALTERNATIVE, status);
204             if (parser.matches()) {
205 
206                 DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, id);
207                 if (deviceSession == null) {
208                     return null;
209                 }
210                 position.setDeviceId(deviceSession.getDeviceId());
211 
212                 position.setNetwork(new Network(CellTower.from(parser.nextInt(0), parser.nextInt(0),
213                         parser.nextHexInt(0), parser.nextHexInt(0), parser.nextInt(0))));
214 
215                 position.set(Position.KEY_BATTERY_LEVEL, parser.nextDouble());
216 
217                 position.set(Position.KEY_FLAGS, parser.next());
218                 position.set(Position.KEY_INPUT, parser.next());
219                 position.set(Position.KEY_OUTPUT, parser.next());
220                 position.set(Position.PREFIX_ADC + 1, parser.next());
221                 position.set(Position.PREFIX_ADC + 2, parser.next());
222                 position.set(Position.PREFIX_ADC + 3, parser.next());
223                 position.set(Position.KEY_ALARM, decodeAlarm(parser.next()));
224 
225             }
226         }
227 
228         return position;
229     }

231     private static final Pattern PATTERN_NEW = new PatternBuilder()
232             .number("dddd").optional()
233             .text("$MGV")
234             .number("ddd,")
235             .number("(d+),")                     // imei
236             .expression("[^,]*,")                // name
237             .expression("([RS]),")
238             .number("(dd)(dd)(dd),")             // date (ddmmyy)
239             .number("(dd)(dd)(dd),")             // time (hhmmss)
240             .expression("([AV]),")               // validity
241             .number("(d+)(dd.d+),([NS]),")       // latitude
242             .number("(d+)(dd.d+),([EW]),")       // longitude
243             .number("dd,")
244             .number("(dd),")                     // satellites
245             .number("dd,")
246             .number("(d+.d+),")                  // hdop
247             .number("(d+.d+)?,")                 // speed
248             .number("(d+.d+)?,")                 // course
249             .number("(-?d+.d+),")                // altitude
250             .number("(d+.d+)?,")                 // odometer
251             .number("(d+),")                     // mcc
252             .number("(d+),")                     // mnc
253             .number("(xxxx),")                   // lac
254             .number("(xxxx),")                   // cid
255             .number("(d+)?,")                    // gsm
256             .expression("([01]+)?,")             // input
257             .expression("([01]+)?,")             // output
258             .number("(d+)?,")                    // adc1
259             .number("(d+)?,")                    // adc2
260             .number("(d+)?,")                    // adc3
261             .groupBegin()
262             .number("(-?d+.?d*)")                // temperature 1
263             .or().text(" ")
264             .groupEnd("?").text(",")
265             .groupBegin()
266             .number("(-?d+.?d*)")                // temperature 2
267             .or().text(" ")
268             .groupEnd("?").text(",")
269             .number("(d+)?,")                    // rfid
270             .number("d*,")
271             .number("(d+)?,")                    // battery
272             .expression("([^,]*);")              // alert
273             .any()
274             .compile();

276     private Position decodeNew(Channel channel, SocketAddress remoteAddress, String sentence) {
277 
278         Parser parser = new Parser(PATTERN_NEW, sentence);
279         if (!parser.matches()) {
280             return null;
281         }
282 
283         Position position = new Position();
284         position.setProtocol(getProtocolName());
285 
286         DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, parser.next());
287         if (deviceSession == null) {
288             return null;
289         }
290         position.setDeviceId(deviceSession.getDeviceId());
291 
292         if (parser.next().equals("S")) {
293             position.set(Position.KEY_ARCHIVE, true);
294         }
295 
296         position.setTime(parser.nextDateTime(Parser.DateTimeFormat.DMY_HMS));
297 
298         position.setValid(parser.next().equals("A"));
299         position.setLatitude(parser.nextCoordinate());
300         position.setLongitude(parser.nextCoordinate());
301 
302         position.set(Position.KEY_SATELLITES, parser.nextInt(0));
303         position.set(Position.KEY_HDOP, parser.nextDouble(0));
304 
305         position.setSpeed(parser.nextDouble(0));
306         position.setCourse(parser.nextDouble(0));
307         position.setAltitude(parser.nextDouble(0));
308 
309         if (parser.hasNext()) {
310             position.set(Position.KEY_ODOMETER, parser.nextDouble(0) * 1000);
311         }
312 
313         position.setNetwork(new Network(CellTower.from(
314                 parser.nextInt(0), parser.nextInt(0), parser.nextHexInt(0), parser.nextHexInt(0), parser.nextInt(0))));
315 
316         position.set(Position.KEY_INPUT, parser.nextBinInt(0));
317         position.set(Position.KEY_OUTPUT, parser.nextBinInt(0));
318 
319         for (int i = 1; i <= 3; i++) {
320             position.set(Position.PREFIX_ADC + i, parser.nextInt(0));
321         }
322 
323         for (int i = 1; i <= 2; i++) {
324             String adc = parser.next();
325             if (adc != null) {
326                 position.set(Position.PREFIX_TEMP + i, Double.parseDouble(adc));
327             }
328         }
329 
330         position.set(Position.KEY_DRIVER_UNIQUE_ID, parser.next());
331 
332         String battery = parser.next();
333         if (battery != null) {
334             position.set(Position.KEY_BATTERY, Integer.parseInt(battery));
335         }
336 
337         position.set(Position.KEY_ALARM, decodeAlarm(parser.next()));
338 
339         return position;
340     }

342     private String decodeAlarm(String value) {
343         switch (value) {
344             case "SOS":
345             case "Help":
346                 return Position.ALARM_SOS;
347             case "Over Speed":
348             case "OverSpeed":
349                 return Position.ALARM_OVERSPEED;
350             case "LowSpeed":
351                 return Position.ALARM_LOW_SPEED;
352             case "Low Battery":
353             case "LowBattery":
354                 return Position.ALARM_LOW_BATTERY;
355             case "VIB":
356                 return Position.ALARM_VIBRATION;
357             case "Move in":
358             case "Geo in":
359             case "Geo1 in":
360             case "Geo2 in":
361                 return Position.ALARM_GEOFENCE_ENTER;
362             case "Move out":
363             case "Geo out":
364             case "Geo1 out":
365             case "Geo2 out":
366                 return Position.ALARM_GEOFENCE_EXIT;
367             default:
368                 return null;
369         }
370     }

372     @Override
373     protected Object decode(
374             Channel channel, SocketAddress remoteAddress, Object msg) throws Exception {
375 
376         String sentence = (String) msg;
377 
378         if (sentence.contains("$MG")) {
379             return decodeNew(channel, remoteAddress, sentence);
380         } else {
381             return decodeOld(channel, remoteAddress, sentence);
382         }
383     }
