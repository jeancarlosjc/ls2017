100     private Position readPosition(DeviceSession deviceSession, ChannelBuffer buf) {
101 
102         Position position = new Position();
103         position.setProtocol(getProtocolName());
104         position.setDeviceId(deviceSession.getDeviceId());
105 
106         DateBuilder dateBuilder = new DateBuilder()
107                 .setDateReverse(buf.readUnsignedByte(), buf.readUnsignedByte(), buf.readUnsignedByte())
108                 .setTime(buf.readUnsignedByte(), buf.readUnsignedByte(), buf.readUnsignedByte());
109         position.setTime(dateBuilder.getDate());
110 
111         double lat = buf.readUnsignedInt() / 3600000.0;
112         double lon = buf.readUnsignedInt() / 3600000.0;
113         position.setSpeed(UnitsConverter.knotsFromCps(buf.readUnsignedShort()));
114         position.setCourse(buf.readUnsignedShort() * 0.1);
115 
116         int flags = buf.readUnsignedByte();
117         if ((flags & 0x02) == 0) {
118             lat = -lat;
119         }
120         if ((flags & 0x01) == 0) {
121             lon = -lon;
122         }
123         position.setLatitude(lat);
124         position.setLongitude(lon);
125         position.setValid((flags & 0x0C) > 0);
126         position.set(Position.KEY_SATELLITES, flags >> 4);
127 
128         return position;
129     }

131     private Position createPosition(DeviceSession deviceSession) {
132 
133         Position position = new Position();
134         position.setProtocol(getProtocolName());
135         position.setDeviceId(deviceSession.getDeviceId());
136 
137         getLastLocation(position, null);
138 
139         return position;
140     }

142     private void decodeObd(Position position, ChannelBuffer buf, boolean groups) {
143 
144         int count = buf.readUnsignedByte();
145 
146         int[] pids = new int[count];
147         for (int i = 0; i < count; i++) {
148             pids[i] = buf.readUnsignedShort() & 0xff;
149         }
150 
151         if (groups) {
152             buf.readUnsignedByte(); // group count
153             buf.readUnsignedByte(); // group size
154         }
155 
156         for (int i = 0; i < count; i++) {
157             int value;
158             switch (PID_LENGTH_MAP.get(pids[i])) {
159                 case 1:
160                     value = buf.readUnsignedByte();
161                     break;
162                 case 2:
163                     value = buf.readUnsignedShort();
164                     break;
165                 case 4:
166                     value = buf.readInt();
167                     break;
168                 default:
169                     value = 0;
170                     break;
171             }
172             position.add(ObdDecoder.decodeData(pids[i], value, false));
173         }
174     }

176     private void decodeStat(Position position, ChannelBuffer buf) {
177 
178         buf.readUnsignedInt(); // ACC ON time
179         buf.readUnsignedInt(); // UTC time
180         position.set(Position.KEY_ODOMETER, buf.readUnsignedInt());
181         position.set(Position.KEY_ODOMETER_TRIP, buf.readUnsignedInt());
182         position.set(Position.KEY_FUEL_CONSUMPTION, buf.readUnsignedInt());
183         buf.readUnsignedShort(); // current fuel consumption
184         position.set(Position.KEY_STATUS, buf.readUnsignedInt());
185         buf.skipBytes(8);
186     }

188     private void sendResponse(
189             Channel channel, SocketAddress remoteAddress,
190             int version, ChannelBuffer id, short type, ChannelBuffer content) {
191 
192         if (channel != null) {
193             int length = 2 + 2 + 1 + id.readableBytes() + 2 + 2 + 2;
194             if (content != null) {
195                 length += content.readableBytes();
196             }
197 
198             ChannelBuffer response = ChannelBuffers.directBuffer(ByteOrder.LITTLE_ENDIAN, length);
199             response.writeByte('@'); response.writeByte('@');
200             response.writeShort(length);
201             response.writeByte(version);
202             response.writeBytes(id);
203             response.writeShort(ChannelBuffers.swapShort(type));
204             if (content != null) {
205                 response.writeBytes(content);
206             }
207             response.writeShort(
208                     Checksum.crc16(Checksum.CRC16_X25, response.toByteBuffer(0, response.writerIndex())));
209             response.writeByte(0x0D); response.writeByte(0x0A);
210             channel.write(response, remoteAddress);
211         }
212     }

214     private void sendResponse(
215             Channel channel, SocketAddress remoteAddress, ChannelBuffer id, short type) {
216 
217         if (channel != null) {
218             int length = 2 + 2 + id.readableBytes() + 2 + 4 + 8 + 2 + 2;
219 
220             ChannelBuffer response = ChannelBuffers.directBuffer(ByteOrder.LITTLE_ENDIAN, length);
221             response.writeByte('@'); response.writeByte('@');
222             response.writeShort(length);
223             response.writeBytes(id);
224             response.writeShort(ChannelBuffers.swapShort(type));
225             response.writeInt(0);
226             for (int i = 0; i < 8; i++) {
227                 response.writeByte(0xff);
228             }
229             response.writeShort(
230                     Checksum.crc16(Checksum.CRC16_X25, response.toByteBuffer(0, response.writerIndex())));
231             response.writeByte(0x0D); response.writeByte(0x0A);
232             channel.write(response, remoteAddress);
233         }
234     }

236     private Object decodeSc(
237             Channel channel, SocketAddress remoteAddress, ChannelBuffer buf,
238             int version, ChannelBuffer id, int type, DeviceSession deviceSession) {
239 
240         if (type == MSG_SC_HEARTBEAT) {
241 
242             sendResponse(channel, remoteAddress, version, id, MSG_SC_HEARTBEAT_RESPONSE, null);
243 
244         } else if (type == MSG_SC_LOGIN || type == MSG_SC_LOGOUT || type == MSG_SC_GPS
245                 || type == MSG_SC_ALARM || type == MSG_SC_CURRENT_LOCATION) {
246 
247             if (type == MSG_SC_LOGIN) {
248                 ChannelBuffer response = ChannelBuffers.directBuffer(ByteOrder.LITTLE_ENDIAN, 10);
249                 response.writeInt(0xFFFFFFFF);
250                 response.writeShort(0);
251                 response.writeInt((int) (System.currentTimeMillis() / 1000));
252                 sendResponse(channel, remoteAddress, version, id, MSG_SC_LOGIN_RESPONSE, response);
253             }
254 
255             if (type == MSG_SC_GPS) {
256                 buf.readUnsignedByte(); // historical
257             } else if (type == MSG_SC_ALARM) {
258                 buf.readUnsignedInt(); // alarm
259             } else if (type == MSG_SC_CURRENT_LOCATION) {
260                 buf.readUnsignedShort();
261             }
262 
263             buf.readUnsignedInt(); // ACC ON time
264             buf.readUnsignedInt(); // UTC time
265             long odometer = buf.readUnsignedInt();
266             long tripOdometer = buf.readUnsignedInt();
267             long fuelConsumption = buf.readUnsignedInt();
268             buf.readUnsignedShort(); // current fuel consumption
269             long status = buf.readUnsignedInt();
270             buf.skipBytes(8);
271 
272             int count = buf.readUnsignedByte();
273 
274             List<Position> positions = new LinkedList<>();
275 
276             for (int i = 0; i < count; i++) {
277                 Position position = readPosition(deviceSession, buf);
278                 position.set(Position.KEY_ODOMETER, odometer);
279                 position.set(Position.KEY_ODOMETER_TRIP, tripOdometer);
280                 position.set(Position.KEY_FUEL_CONSUMPTION, fuelConsumption);
281                 position.set(Position.KEY_STATUS, status);
282                 positions.add(position);
283             }
284 
285             if (!positions.isEmpty()) {
286                 return positions;
287             }
288 
289         } else if (type == MSG_SC_GPS_SLEEP) {
290 
291             buf.readUnsignedInt(); // device time
292 
293             return readPosition(deviceSession, buf);
294 
295         } else if (type == MSG_SC_AGPS_REQUEST) {
296 
297             return readPosition(deviceSession, buf);
298 
299         } else if (type == MSG_SC_PID_DATA) {
300 
301             Position position = createPosition(deviceSession);
302 
303             decodeStat(position, buf);
304 
305             buf.readUnsignedShort(); // sample rate
306             decodeObd(position, buf, true);
307 
308             return position;
309 
310         } else if (type == MSG_SC_DTCS_PASSENGER) {
311 
312             Position position = createPosition(deviceSession);
313 
314             decodeStat(position, buf);
315 
316             buf.readUnsignedByte(); // flag
317             position.add(ObdDecoder.decodeCodes(ChannelBuffers.hexDump(buf.readBytes(buf.readUnsignedByte()))));
318 
319             return position;
320 
321         } else if (type == MSG_SC_OBD_DATA) {
322 
323             Position position = createPosition(deviceSession);
324 
325             decodeStat(position, buf);
326 
327             buf.readUnsignedByte(); // flag
328             decodeObd(position, buf, false);
329 
330             return position;
331 
332         } else if (type == MSG_SC_CELL) {
333 
334             Position position = createPosition(deviceSession);
335 
336             decodeStat(position, buf);
337 
338             position.setNetwork(new Network(
339                     CellTower.fromLacCid(buf.readUnsignedShort(), buf.readUnsignedShort())));
340 
341             return position;
342 
343         }
344 
345         return null;
346     }

349     private Object decodeCc(
350             Channel channel, SocketAddress remoteAddress, ChannelBuffer buf,
351             int version, ChannelBuffer id, int type, DeviceSession deviceSession) {
352 
353         if (type == MSG_CC_HEARTBEAT) {
354 
355             sendResponse(channel, remoteAddress, version, id, MSG_CC_HEARTBEAT_RESPONSE, null);
356 
357             buf.readUnsignedByte(); // 0x01 for history
358             int count = buf.readUnsignedByte();
359 
360             List<Position> positions = new LinkedList<>();
361 
362             for (int i = 0; i < count; i++) {
363                 Position position = readPosition(deviceSession, buf);
364 
365                 position.set(Position.KEY_STATUS, buf.readUnsignedInt());
366                 position.set(Position.KEY_BATTERY, buf.readUnsignedByte());
367                 position.set(Position.KEY_ODOMETER, buf.readUnsignedInt());
368 
369                 buf.readUnsignedByte(); // geo-fencing id
370                 buf.readUnsignedByte(); // geo-fencing flags
371                 buf.readUnsignedByte(); // additional flags
372 
373                 position.setNetwork(new Network(
374                         CellTower.fromLacCid(buf.readUnsignedShort(), buf.readUnsignedShort())));
375 
376                 positions.add(position);
377             }
378 
379             return positions;
380 
381         } else if (type == MSG_CC_LOGIN) {
382 
383             sendResponse(channel, remoteAddress, version, id, MSG_CC_LOGIN_RESPONSE, null);
384 
385             Position position = readPosition(deviceSession, buf);
386 
387             position.set(Position.KEY_STATUS, buf.readUnsignedInt());
388             position.set(Position.KEY_BATTERY, buf.readUnsignedByte());
389             position.set(Position.KEY_ODOMETER, buf.readUnsignedInt());
390 
391             buf.readUnsignedByte(); // geo-fencing id
392             buf.readUnsignedByte(); // geo-fencing flags
393             buf.readUnsignedByte(); // additional flags
394 
395             // GSM_CELL_CODE
396             // STR_Z - firmware version
397             // STR_Z - hardware version
398 
399             return position;
400 
401         }
402 
403         return null;
404     }

406     @Override
407     protected Object decode(
408             Channel channel, SocketAddress remoteAddress, Object msg) throws Exception {
409 
410         ChannelBuffer buf = (ChannelBuffer) msg;
411 
412         int header = buf.readUnsignedShort();
413         buf.readUnsignedShort(); // length
414 
415         int version = -1;
416         if (header == 0x4040) {
417             version = buf.readUnsignedByte();
418         }
419 
420         ChannelBuffer id = buf.readBytes(20);
421         int type = ChannelBuffers.swapShort(buf.readShort());
422 
423         DeviceSession deviceSession = getDeviceSession(
424                 channel, remoteAddress, id.toString(StandardCharsets.US_ASCII).trim());
425         if (deviceSession == null) {
426             return null;
427         }
428 
429         if (version == -1) {
430 
431             if (type == 0x2001) {
432 
433                 sendResponse(channel, remoteAddress, id, (short) 0x1001);
434 
435                 buf.readUnsignedInt(); // index
436                 buf.readUnsignedInt(); // unix time
437                 buf.readUnsignedByte();
438 
439                 return readPosition(deviceSession, buf);
440 
441             }
442 
443         } else if (version == 3 || version == 4) {
444 
445             return decodeSc(channel, remoteAddress, buf, version, id, type, deviceSession);
446 
447         } else {
448 
449             return decodeCc(channel, remoteAddress, buf, version, id, type, deviceSession);
450 
451         }
452 
453         return null;
454     }
