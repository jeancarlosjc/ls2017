17  package org.traccar.protocol;
18  
19  import org.traccar.StringProtocolEncoder;
20  import org.traccar.helper.Log;
21  import org.traccar.model.Command;
22  
23  public class Tk103ProtocolEncoder extends StringProtocolEncoder {
24  
25      @Override
26      protected Object encodeCommand(Command command) {
27  
28          switch (command.getType()) {
29              case Command.TYPE_GET_VERSION:
30                  return formatCommand(command, "({%s}AP07)", Command.KEY_UNIQUE_ID);
31              case Command.TYPE_REBOOT_DEVICE:
32                  return formatCommand(command, "({%s}AT00)", Command.KEY_UNIQUE_ID);
33              case Command.TYPE_SET_ODOMETER:
34                  return formatCommand(command, "({%s}AX01)", Command.KEY_UNIQUE_ID);
35              case Command.TYPE_POSITION_SINGLE:
36                  return formatCommand(command, "({%s}AP00)", Command.KEY_UNIQUE_ID);
37              case Command.TYPE_POSITION_PERIODIC:
38                  return formatCommand(command, "({%s}AR00%s0000)", Command.KEY_UNIQUE_ID,
39                          String.format("%04X", command.getInteger(Command.KEY_FREQUENCY)));
40              case Command.TYPE_POSITION_STOP:
41                  return formatCommand(command, "({%s}AR0000000000)", Command.KEY_UNIQUE_ID);
42              case Command.TYPE_ENGINE_STOP:
43                  return formatCommand(command, "({%s}AV011)", Command.KEY_UNIQUE_ID);
44              case Command.TYPE_ENGINE_RESUME:
45                  return formatCommand(command, "({%s}AV010)", Command.KEY_UNIQUE_ID);
46              default:
47                  Log.warning(new UnsupportedOperationException(command.getType()));
48                  break;
49          }
50  
51          return null;
52      }
53  
54  }
