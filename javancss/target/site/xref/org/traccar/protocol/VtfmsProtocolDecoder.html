16  package org.traccar.protocol;
17  
18  import org.jboss.netty.channel.Channel;
19  import org.traccar.BaseProtocolDecoder;
20  import org.traccar.DeviceSession;
21  import org.traccar.helper.Parser;
22  import org.traccar.helper.PatternBuilder;
23  import org.traccar.helper.UnitsConverter;
24  import org.traccar.model.Position;
25  
26  import java.net.SocketAddress;
27  import java.util.regex.Pattern;
28  
29  public class VtfmsProtocolDecoder extends BaseProtocolDecoder {
30  
31      private static final String[] DIRECTIONS = new String[] {"N", "NE", "E", "SE", "S", "SW", "W", "NW"};
32  
33      public VtfmsProtocolDecoder(VtfmsProtocol protocol) {
34          super(protocol);
35      }
36  
37      private static final Pattern PATTERN = new PatternBuilder()
38              .text("(")
39              .number("(d{15}),")                  // imei
40              .number("[0-9A-Z]{3}dd,")            // packet count
41              .number("(dd),")                     // packet id
42              .number("[^,]*,")                    // reserved
43              .number("(d+)?,")                    // rssi
44              .number("(?:d+)?,")                  // fix status
45              .number("(d+)?,")                    // satellites
46              .number("[^,]*,")                    // reserved
47              .expression("([AV]),")               // validity
48              .number("(dd)(dd)(dd),")             // time (hhmmss)
49              .number("(dd)(dd)(dd),")             // time (ddmmyy)
50              .number("(-?d+.d+),")                // latitude
51              .number("(-?d+.d+),")                // longitude
52              .number("(?:(d+)|([NESW]{1,2})),")   // course
53              .number("(d+),")                     // speed
54              .number("(d+),")                     // hours
55              .number("(d+),")                     // idle hours
56              .expression("[KNT],")                // antenna status
57              .number("(d+),")                     // odometer
58              .expression("([01]),")               // power status
59              .number("(d+.d+),")                  // power voltage
60              .number("[^,]*,")                    // reserved
61              .number("(d+)?,")                    // fuel level
62              .number("(d+.d+),")                  // adc 1
63              .number("[^,]*,")                    // reserved
64              .number("(d+.d+)?,")                 // adc 2
65              .expression("([01]),")               // di 1
66              .expression("([01]),")               // di 2
67              .expression("([01]),")               // di 3
68              .expression("([01]),")               // di 4
69              .expression("([01]),")               // do 1
70              .expression("([01]),")               // do 2
71              .expression("([01]),")               // do 3
72              .number("[^,]*,")                    // reserved
73              .number("[^,]*")                     // reserved
74              .text(")")
75              .number("ddd")                       // checksum
76              .compile();
77  
78      private String decodeAlarm(int value) {
79          switch (value) {
80              case 10:
81                  return Position.ALARM_OVERSPEED;
82              case 14:
83                  return Position.ALARM_POWER_CUT;
84              case 15:
85                  return Position.ALARM_POWER_RESTORED;
86              case 32:
87                  return Position.ALARM_BRAKING;
88              case 33:
89                  return Position.ALARM_ACCELERATION;
90              default:
91                  return null;
92          }
93      }
94  
95      @Override
96      protected Object decode(
97              Channel channel, SocketAddress remoteAddress, Object msg) throws Exception {
98  
99          Parser parser = new Parser(PATTERN, (String) msg);
100         if (!parser.matches()) {
101             return null;
102         }
103 
104         DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, parser.next());
105         if (deviceSession == null) {
106             return null;
107         }
108 
109         Position position = new Position();
110         position.setProtocol(getProtocolName());
111         position.setDeviceId(deviceSession.getDeviceId());
112 
113         position.set(Position.KEY_ALARM, decodeAlarm(parser.nextInt()));
114         position.set(Position.KEY_RSSI, parser.nextInt());
115         position.set(Position.KEY_SATELLITES, parser.nextInt());
116 
117         position.setValid(parser.next().equals("A"));
118         position.setTime(parser.nextDateTime(Parser.DateTimeFormat.HMS_DMY));
119         position.setLatitude(parser.nextDouble(0));
120         position.setLongitude(parser.nextDouble(0));
121 
122         if (parser.hasNext()) {
123             position.setCourse(parser.nextDouble(0));
124         }
125         if (parser.hasNext()) {
126             String direction = parser.next();
127             for (int i = 0; i < DIRECTIONS.length; i++) {
128                 if (direction.equals(DIRECTIONS[i])) {
129                     position.setCourse(i * 45.0);
130                     break;
131                 }
132             }
133         }
134 
135         position.setSpeed(UnitsConverter.knotsFromKph(parser.nextDouble(0)));
136 
137         position.set(Position.KEY_HOURS, parser.nextInt());
138         position.set("idleHours", parser.nextInt());
139         position.set(Position.KEY_ODOMETER, parser.nextInt() * 100);
140         position.set(Position.KEY_CHARGE, parser.next().equals("1"));
141         position.set(Position.KEY_POWER, parser.nextDouble());
142         position.set(Position.KEY_FUEL_LEVEL, parser.nextInt());
143         position.set(Position.PREFIX_ADC + 1, parser.nextDouble());
144         position.set(Position.PREFIX_ADC + 2, parser.nextDouble());
145         position.set(Position.PREFIX_IN + 1, parser.nextInt());
146         position.set(Position.PREFIX_IN + 2, parser.nextInt());
147         position.set(Position.PREFIX_IN + 3, parser.nextInt());
148         position.set(Position.PREFIX_IN + 4, parser.nextInt());
149         position.set(Position.PREFIX_OUT + 1, parser.nextInt());
150         position.set(Position.PREFIX_OUT + 2, parser.nextInt());
151         position.set(Position.PREFIX_OUT + 3, parser.nextInt());
152 
153         return position;
154     }
155 
156 }
