341     private Object decodeAck(Channel channel, SocketAddress remoteAddress, String sentence, String type) {
342         Parser parser = new Parser(PATTERN_ACK, sentence);
343         if (parser.matches()) {
344             String protocolVersion = parser.next();
345             DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, parser.next());
346             if (deviceSession == null) {
347                 return null;
348             }
349             if (type.equals("HBD")) {
350                 if (channel != null) {
351                     parser.skip(6);
352                     channel.write("+SACK:GTHBD," + protocolVersion + "," + parser.next() + "$", remoteAddress);
353                 }
354             } else {
355                 Position position = new Position();
356                 position.setProtocol(getProtocolName());
357                 position.setDeviceId(deviceSession.getDeviceId());
358                 getLastLocation(position, parser.nextDateTime());
359                 position.setValid(false);
360                 position.set(Position.KEY_RESULT, "Command " + type + " accepted");
361                 return position;
362             }
363         }
364         return null;
365     }

367     private Position initPosition(Parser parser, Channel channel, SocketAddress remoteAddress) {
368         if (parser.matches()) {
369             DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, parser.next());
370             if (deviceSession != null) {
371                 Position position = new Position();
372                 position.setProtocol(getProtocolName());
373                 position.setDeviceId(deviceSession.getDeviceId());
374                 return position;
375             }
376         }
377         return null;
378     }

390     private Object decodeInf(Channel channel, SocketAddress remoteAddress, String sentence) {
391         Parser parser = new Parser(PATTERN_INF, sentence);
392         Position position = initPosition(parser, channel, remoteAddress);
393         if (position == null) {
394             return null;
395         }
396 
397         switch (parser.nextHexInt()) {
398             case 0x16:
399             case 0x1A:
400             case 0x12:
401                 position.set(Position.KEY_IGNITION, false);
402                 position.set(Position.KEY_MOTION, true);
403                 break;
404             case 0x11:
405                 position.set(Position.KEY_IGNITION, false);
406                 position.set(Position.KEY_MOTION, false);
407                 break;
408             case 0x21:
409                 position.set(Position.KEY_IGNITION, true);
410                 position.set(Position.KEY_MOTION, false);
411                 break;
412             case 0x22:
413                 position.set(Position.KEY_IGNITION, true);
414                 position.set(Position.KEY_MOTION, true);
415                 break;
416             case 0x41:
417                 position.set(Position.KEY_MOTION, false);
418                 break;
419             case 0x42:
420                 position.set(Position.KEY_MOTION, true);
421                 break;
422             default:
423                 break;
424         }
425 
426         position.set(Position.KEY_RSSI, parser.nextInt());
427 
428         parser.next(); // odometer or external power
429 
430         position.set(Position.KEY_BATTERY, parser.nextDouble(0));
431         position.set(Position.KEY_CHARGE, parser.nextInt(0) == 1);
432 
433         position.set(Position.KEY_BATTERY_LEVEL, parser.nextInt());
434 
435         position.set(Position.PREFIX_TEMP + 1, parser.next());
436 
437         position.set(Position.PREFIX_ADC + 1, parser.next());
438         position.set(Position.PREFIX_ADC + 2, parser.next());
439 
440         position.set(Position.KEY_INPUT, parser.next());
441         position.set(Position.KEY_OUTPUT, parser.next());
442 
443         getLastLocation(position, parser.nextDateTime());
444 
445         position.set(Position.KEY_INDEX, parser.nextHexInt(0));
446 
447         return position;
448     }

450     private Object decodeVer(Channel channel, SocketAddress remoteAddress, String sentence) {
451         Parser parser = new Parser(PATTERN_VER, sentence);
452         Position position = initPosition(parser, channel, remoteAddress);
453         if (position == null) {
454             return null;
455         }
456 
457         position.set("deviceType", parser.next());
458         position.set(Position.KEY_VERSION_FW, parser.nextHexInt(0));
459         position.set(Position.KEY_VERSION_HW, parser.nextHexInt(0));
460 
461         getLastLocation(position, parser.nextDateTime());
462 
463         return position;
464     }

466     private void decodeLocation(Position position, Parser parser) {
467         int hdop = parser.nextInt(0);
468         position.setValid(hdop > 0);
469         position.set(Position.KEY_HDOP, hdop);
470 
471         position.setSpeed(UnitsConverter.knotsFromKph(parser.nextDouble(0)));
472         position.setCourse(parser.nextDouble(0));
473         position.setAltitude(parser.nextDouble(0));
474 
475         if (parser.hasNext(8)) {
476             position.setValid(true);
477             position.setLongitude(parser.nextDouble(0));
478             position.setLatitude(parser.nextDouble(0));
479             position.setTime(parser.nextDateTime());
480         } else {
481             getLastLocation(position, null);
482         }
483 
484         if (parser.hasNext(6)) {
485             int mcc = parser.nextInt(0);
486             int mnc = parser.nextInt(0);
487             if (parser.hasNext(2)) {
488                 position.setNetwork(new Network(CellTower.from(mcc, mnc, parser.nextInt(0), parser.nextInt(0))));
489             }
490             if (parser.hasNext(2)) {
491                 position.setNetwork(new Network(CellTower.from(mcc, mnc, parser.nextHexInt(0), parser.nextHexInt(0))));
492             }
493         }
494 
495         position.set(Position.KEY_ODOMETER, parser.nextDouble(0) * 1000);
496     }

498     private Object decodeObd(Channel channel, SocketAddress remoteAddress, String sentence) {
499         Parser parser = new Parser(PATTERN_OBD, sentence);
500         Position position = initPosition(parser, channel, remoteAddress);
501         if (position == null) {
502             return null;
503         }
504 
505         position.set(Position.KEY_RPM, parser.nextInt());
506         position.set(Position.KEY_OBD_SPEED, parser.nextInt());
507         position.set(Position.PREFIX_TEMP + 1, parser.nextInt());
508         position.set(Position.KEY_FUEL_CONSUMPTION, parser.next());
509         position.set("dtcsClearedDistance", parser.nextInt());
510         position.set("odbConnect", parser.nextInt(0) == 1);
511         position.set("dtcsNumber", parser.nextInt());
512         position.set("dtcsCodes", parser.next());
513         position.set(Position.KEY_THROTTLE, parser.nextInt());
514         position.set(Position.KEY_FUEL_LEVEL, parser.nextInt());
515         position.set(Position.KEY_OBD_ODOMETER, parser.nextInt(0) * 1000);
516 
517         decodeLocation(position, parser);
518 
519         position.set(Position.KEY_ODOMETER, parser.nextDouble(0) * 1000);
520 
521         decodeDeviceTime(position, parser);
522 
523         return position;
524     }

526     private void decodeStatus(Position position, Parser parser) {
527         if (parser.hasNext(3)) {
528             int ignition = parser.nextHexInt(0);
529             if (BitUtil.check(ignition, 4)) {
530                 position.set(Position.KEY_IGNITION, false);
531             } else if (BitUtil.check(ignition, 5)) {
532                 position.set(Position.KEY_IGNITION, true);
533             }
534             position.set(Position.KEY_INPUT, parser.nextHexInt(0));
535             position.set(Position.KEY_OUTPUT, parser.nextHexInt(0));
536         }
537     }

539     private Object decodeFri(Channel channel, SocketAddress remoteAddress, String sentence) {
540         Parser parser = new Parser(PATTERN_FRI, sentence);
541         if (!parser.matches()) {
542             return null;
543         }
544 
545         DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, parser.next());
546         if (deviceSession == null) {
547             return null;
548         }
549 
550         LinkedList<Position> positions = new LinkedList<>();
551 
552         String vin = parser.next();
553         int power = parser.nextInt(0);
554 
555         Parser itemParser = new Parser(PATTERN_LOCATION, parser.next());
556         while (itemParser.find()) {
557             Position position = new Position();
558             position.setProtocol(getProtocolName());
559             position.setDeviceId(deviceSession.getDeviceId());
560 
561             position.set(Position.KEY_VIN, vin);
562 
563             decodeLocation(position, itemParser);
564 
565             positions.add(position);
566         }
567 
568         Position position = positions.getLast();
569 
570         decodeLocation(position, parser);
571 
572         // power value only on some devices
573         if (power > 10) {
574             position.set(Position.KEY_POWER, power);
575         }
576 
577         position.set(Position.KEY_ODOMETER, parser.nextDouble(0) * 1000);
578         position.set(Position.KEY_BATTERY_LEVEL, parser.nextInt());
579 
580         position.set(Position.KEY_ODOMETER, parser.nextDouble(0) * 1000);
581         position.set(Position.KEY_HOURS, parser.next());
582         position.set(Position.PREFIX_ADC + 1, parser.next());
583         position.set(Position.PREFIX_ADC + 2, parser.next());
584         position.set(Position.KEY_BATTERY_LEVEL, parser.nextInt());
585 
586         decodeStatus(position, parser);
587 
588         position.set(Position.KEY_RPM, parser.nextInt());
589         position.set(Position.KEY_FUEL_LEVEL, parser.nextInt());
590 
591         decodeDeviceTime(position, parser);
592         if (ignoreFixTime) {
593             positions.clear();
594             positions.add(position);
595         }
596 
597         return positions;
598     }

600     private Object decodeEri(Channel channel, SocketAddress remoteAddress, String sentence) {
601         Parser parser = new Parser(PATTERN_ERI, sentence);
602         if (!parser.matches()) {
603             return null;
604         }
605 
606         DeviceSession deviceSession = getDeviceSession(channel, remoteAddress, parser.next());
607         if (deviceSession == null) {
608             return null;
609         }
610 
611         LinkedList<Position> positions = new LinkedList<>();
612 
613         int power = parser.nextInt(0);
614 
615         Parser itemParser = new Parser(PATTERN_LOCATION, parser.next());
616         while (itemParser.find()) {
617             Position position = new Position();
618             position.setProtocol(getProtocolName());
619             position.setDeviceId(deviceSession.getDeviceId());
620 
621             decodeLocation(position, itemParser);
622 
623             positions.add(position);
624         }
625 
626         Position position = positions.getLast();
627 
628         decodeLocation(position, parser);
629 
630         position.set(Position.KEY_POWER, power);
631         position.set(Position.KEY_ODOMETER, parser.nextDouble(0) * 1000);
632         position.set(Position.KEY_HOURS, parser.next());
633         position.set(Position.PREFIX_ADC + 1, parser.next());
634         position.set(Position.PREFIX_ADC + 2, parser.next());
635         position.set(Position.KEY_BATTERY_LEVEL, parser.nextInt());
636 
637         decodeStatus(position, parser);
638 
639         int index = 0;
640         String[] data = parser.next().split(",");
641         if (data.length > 1) {
642             int deviceType = Integer.parseInt(data[index++]);
643             if (deviceType == 2) {
644                 int deviceCount = Integer.parseInt(data[index++]);
645                 for (int i = 1; i <= deviceCount; i++) {
646                     index++; // id
647                     index++; // type
648                     position.set(Position.PREFIX_TEMP + i, (short) Integer.parseInt(data[index++], 16) * 0.0625);
649                 }
650             }
651         }
652 
653         decodeDeviceTime(position, parser);
654         if (ignoreFixTime) {
655             positions.clear();
656             positions.add(position);
657         }
658 
659         return positions;
660     }

697     private Object decodeWif(Channel channel, SocketAddress remoteAddress, String sentence) {
698         Parser parser = new Parser(PATTERN_WIF, sentence);
699         Position position = initPosition(parser, channel, remoteAddress);
700         if (position == null) {
701             return null;
702         }
703 
704         getLastLocation(position, null);
705 
706         Network network = new Network();
707 
708         parser.nextInt(0); // count
709         Matcher matcher = Pattern.compile("([0-9a-fA-F]{12}),(-?\\d+),,,,").matcher(parser.next());
710         while (matcher.find()) {
711             String mac = matcher.group(1).replaceAll("(..)", "$1:");
712             network.addWifiAccessPoint(WifiAccessPoint.from(
713                     mac.substring(0, mac.length() - 1), Integer.parseInt(matcher.group(2))));
714         }
715 
716         position.setNetwork(network);
717 
718         position.set(Position.KEY_BATTERY_LEVEL, parser.nextInt(0));
719 
720         return position;
721     }

742         }

749     private Object decodeOther(Channel channel, SocketAddress remoteAddress, String sentence, String type) {
750         Parser parser = new Parser(PATTERN, sentence);
751         Position position = initPosition(parser, channel, remoteAddress);
752         if (position == null) {
753             return null;
754         }
755 
756         int reportType = parser.nextInt(0);
757         if (type.equals("NMR")) {
758             position.set(Position.KEY_MOTION, reportType == 1);
759         } else if (type.equals("SOS")) {
760             position.set(Position.KEY_ALARM, Position.ALARM_SOS);
761         }
762 
763         decodeLocation(position, parser);
764 
765         position.set(Position.KEY_ODOMETER, parser.nextDouble(0) * 1000);
766         position.set(Position.KEY_BATTERY_LEVEL, parser.nextInt(0));
767 
768         position.set(Position.KEY_ODOMETER, parser.nextDouble(0) * 1000);
769 
770         decodeDeviceTime(position, parser);
771 
772         if (Context.getConfig().getBoolean(getProtocolName() + ".ack") && channel != null) {
773             channel.write("+SACK:" + parser.next() + "$", remoteAddress);
774         }
775 
776         return position;
777     }

779     private Object decodeBasic(Channel channel, SocketAddress remoteAddress, String sentence, String type) {
780         Parser parser = new Parser(PATTERN_BASIC, sentence);
781         Position position = initPosition(parser, channel, remoteAddress);
782         if (position == null) {
783             return null;
784         }
785 
786         int hdop = parser.nextInt(0);
787         position.setValid(hdop > 0);
788         position.set(Position.KEY_HDOP, hdop);
789 
790         position.setSpeed(UnitsConverter.knotsFromKph(parser.nextDouble(0)));
791         position.setCourse(parser.nextDouble(0));
792         position.setAltitude(parser.nextDouble(0));
793 
794         if (parser.hasNext(2)) {
795             position.setLongitude(parser.nextDouble(0));
796             position.setLatitude(parser.nextDouble(0));
797         } else {
798             getLastLocation(position, null);
799         }
800 
801         if (parser.hasNext(6)) {
802             position.setTime(parser.nextDateTime());
803         }
804 
805         if (parser.hasNext(4)) {
806             position.setNetwork(new Network(CellTower.from(
807                     parser.nextInt(0), parser.nextInt(0), parser.nextHexInt(0), parser.nextHexInt(0))));
808         }
809 
810         decodeDeviceTime(position, parser);
811 
812         switch (type) {
813             case "PNA":
814                 position.set(Position.KEY_ALARM, Position.ALARM_POWER_ON);
815                 break;
816             case "PFA":
817                 position.set(Position.KEY_ALARM, Position.ALARM_POWER_OFF);
818                 break;
819             case "EPN":
820                 position.set(Position.KEY_ALARM, Position.ALARM_POWER_RESTORED);
821                 break;
822             case "EPF":
823                 position.set(Position.KEY_ALARM, Position.ALARM_POWER_CUT);
824                 break;
825             case "BPL":
826                 position.set(Position.KEY_ALARM, Position.ALARM_LOW_BATTERY);
827                 break;
828             case "STT":
829                 position.set(Position.KEY_ALARM, Position.ALARM_MOVEMENT);
830                 break;
831             case "SWG":
832                 position.set(Position.KEY_ALARM, Position.ALARM_GEOFENCE);
833                 break;
834             case "TMP":
835             case "TEM":
836                 position.set(Position.KEY_ALARM, Position.ALARM_TEMPERATURE);
837                 break;
838             case "JDR":
839             case "JDS":
840                 position.set(Position.KEY_ALARM, Position.ALARM_JAMMING);
841                 break;
842             default:
843                 break;
844         }
845 
846         return position;
847     }

849     @Override
850     protected Object decode(
851             Channel channel, SocketAddress remoteAddress, Object msg) throws Exception {
852 
853         String sentence = ((ChannelBuffer) msg).toString(StandardCharsets.US_ASCII);
854 
855         int typeIndex = sentence.indexOf(":GT");
856         if (typeIndex < 0) {
857             return null;
858         }
859 
860         Object result;
861         String type = sentence.substring(typeIndex + 3, typeIndex + 6);
862         if (sentence.startsWith("+ACK")) {
863             result = decodeAck(channel, remoteAddress, sentence, type);
864         } else {
865             switch (type) {
866                 case "INF":
867                     result = decodeInf(channel, remoteAddress, sentence);
868                     break;
869                 case "OBD":
870                     result = decodeObd(channel, remoteAddress, sentence);
871                     break;
872                 case "FRI":
873                     result = decodeFri(channel, remoteAddress, sentence);
874                     break;
875                 case "ERI":
876                     result = decodeEri(channel, remoteAddress, sentence);
877                     break;
878                 case "IGN":
879                 case "IGF":
880                     result = decodeIgn(channel, remoteAddress, sentence);
881                     break;
882                 case "IDA":
883                     result = decodeIda(channel, remoteAddress, sentence);
884                     break;
885                 case "WIF":
886                     result = decodeWif(channel, remoteAddress, sentence);
887                     break;
888                 case "GSM":
889                     result = decodeGsm(channel, remoteAddress, sentence);
890                     break;
891                 case "VER":
892                     result = decodeVer(channel, remoteAddress, sentence);
893                     break;
894                 default:
895                     result = decodeOther(channel, remoteAddress, sentence, type);
896                     break;
897             }
898 
899             if (result == null) {
900                 result = decodeBasic(channel, remoteAddress, sentence, type);
901             }
902 
903             if (result != null) {
904                 if (result instanceof Position) {
905                     ((Position) result).set(Position.KEY_TYPE, type);
906                 } else {
907                     for (Position p : (List<Position>) result) {
908                         p.set(Position.KEY_TYPE, type);
909                     }
910                 }
911             }
912         }
913 
914         return result;
915     }
