16  package org.traccar.helper;
17  
18  import java.util.Calendar;
19  import java.util.Date;
20  
21  import org.joda.time.format.DateTimeFormatter;
22  import org.joda.time.format.ISODateTimeFormat;
23  
24  public final class DateUtil {
25  
26      private static final DateTimeFormatter DATE_FORMAT = ISODateTimeFormat.dateTimeParser();
27  
28      private DateUtil() {
29      }
30  
31      public static Date correctDay(Date guess) {
32          return correctDate(new Date(), guess, Calendar.DAY_OF_MONTH);
33      }
34  
35      public static Date correctYear(Date guess) {
36          return correctDate(new Date(), guess, Calendar.YEAR);
37      }
38  
39      public static Date correctDate(Date now, Date guess, int field) {
40  
41          if (guess.getTime() > now.getTime()) {
42              Date previous = dateAdd(guess, field, -1);
43              if (now.getTime() - previous.getTime() < guess.getTime() - now.getTime()) {
44                  return previous;
45              }
46          } else if (guess.getTime() < now.getTime()) {
47              Date next = dateAdd(guess, field, 1);
48              if (next.getTime() - now.getTime() < now.getTime() - guess.getTime()) {
49                  return next;
50              }
51          }
52  
53          return guess;
54      }
55  
56      private static Date dateAdd(Date guess, int field, int amount) {
57          Calendar calendar = Calendar.getInstance();
58          calendar.setTime(guess);
59          calendar.add(field, amount);
60          return calendar.getTime();
61      }
62  
63      public static Date parseDate(String value) {
64          return DATE_FORMAT.parseDateTime(value).toDate();
65      }
66  }
