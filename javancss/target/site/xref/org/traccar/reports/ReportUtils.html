57      public static void checkPeriodLimit(Date from, Date to) {
58          long limit = Context.getConfig().getLong("report.periodLimit") * 1000;
59          if (limit > 0 && to.getTime() - from.getTime() > limit) {
60              throw new IllegalArgumentException("Time period exceeds the limit");
61          }
62      }

94      public static double calculateDistance(Position firstPosition, Position lastPosition, boolean useOdometer) {
95          double distance = 0.0;
96          double firstOdometer = firstPosition.getDouble(Position.KEY_ODOMETER);
97          double lastOdometer = lastPosition.getDouble(Position.KEY_ODOMETER);
98  
99          if (useOdometer && (firstOdometer != 0.0 || lastOdometer != 0.0)) {
100             distance = lastOdometer - firstOdometer;
101         } else if (firstPosition.getAttributes().containsKey(Position.KEY_TOTAL_DISTANCE)
102                 && lastPosition.getAttributes().containsKey(Position.KEY_TOTAL_DISTANCE)) {
103             distance = lastPosition.getDouble(Position.KEY_TOTAL_DISTANCE)
104                     - firstPosition.getDouble(Position.KEY_TOTAL_DISTANCE);
105         }
106         return distance;
107     }

109     public static double calculateFuel(Position firstPosition, Position lastPosition) {
110 
111         if (firstPosition.getAttributes().get(Position.KEY_FUEL_LEVEL) != null
112                 && lastPosition.getAttributes().get(Position.KEY_FUEL_LEVEL) != null) {
113 
114             BigDecimal value = new BigDecimal(firstPosition.getDouble(Position.KEY_FUEL_LEVEL)
115                     - lastPosition.getDouble(Position.KEY_FUEL_LEVEL));
116             return value.setScale(1, RoundingMode.HALF_EVEN).doubleValue();
117         }
118         return 0;
119     }

121     public static String findDriver(Position firstPosition, Position lastPosition) {
122         if (firstPosition.getAttributes().containsKey(Position.KEY_DRIVER_UNIQUE_ID)) {
123             return firstPosition.getString(Position.KEY_DRIVER_UNIQUE_ID);
124         } else if (lastPosition.getAttributes().containsKey(Position.KEY_DRIVER_UNIQUE_ID)) {
125             return lastPosition.getString(Position.KEY_DRIVER_UNIQUE_ID);
126         }
127         return null;
128     }

140     public static org.jxls.common.Context initializeContext(long userId) {
141         org.jxls.common.Context jxlsContext = PoiTransformer.createInitialContext();
142         jxlsContext.putVar("distanceUnit", getDistanceUnit(userId));
143         jxlsContext.putVar("speedUnit", getSpeedUnit(userId));
144         jxlsContext.putVar("volumeUnit", getVolumeUnit(userId));
145         jxlsContext.putVar("webUrl", Context.getVelocityEngine().getProperty("web.url"));
146         jxlsContext.putVar("dateTool", new DateTool());
147         jxlsContext.putVar("numberTool", new NumberTool());
148         jxlsContext.putVar("timezone", getTimezone(userId));
149         jxlsContext.putVar("locale", Locale.getDefault());
150         jxlsContext.putVar("bracketsRegex", "[\\{\\}\"]");
151         return jxlsContext;
152     }

154     public static void processTemplateWithSheets(InputStream templateStream, OutputStream targetStream,
155             org.jxls.common.Context jxlsContext) throws IOException {
156         Transformer transformer = TransformerFactory.createTransformer(templateStream, targetStream);
157         List<Area> xlsAreas = new XlsCommentAreaBuilder(transformer).build();
158         for (Area xlsArea : xlsAreas) {
159             xlsArea.applyAt(new CellRef(xlsArea.getStartCellRef().getCellName()), jxlsContext);
160             xlsArea.setFormulaProcessor(new StandardFormulaProcessor());
161             xlsArea.processFormulas();
162         }
163         transformer.deleteSheet(xlsAreas.get(0).getStartCellRef().getSheetName());
164         transformer.write();
165     }

167     private static TripReport calculateTrip(
168             ArrayList<Position> positions, int startIndex, int endIndex, boolean ignoreOdometer) {
169         Position startTrip = positions.get(startIndex);
170         Position endTrip = positions.get(endIndex);
171 
172         double speedMax = 0.0;
173         double speedSum = 0.0;
174         for (int i = startIndex; i <= endIndex; i++) {
175             double speed = positions.get(i).getSpeed();
176             speedSum += speed;
177             if (speed > speedMax) {
178                 speedMax = speed;
179             }
180         }
181 
182         TripReport trip = new TripReport();
183 
184         long tripDuration = endTrip.getFixTime().getTime() - startTrip.getFixTime().getTime();
185         long deviceId = startTrip.getDeviceId();
186         trip.setDeviceId(deviceId);
187         trip.setDeviceName(Context.getIdentityManager().getById(deviceId).getName());
188 
189         trip.setStartPositionId(startTrip.getId());
190         trip.setStartLat(startTrip.getLatitude());
191         trip.setStartLon(startTrip.getLongitude());
192         trip.setStartTime(startTrip.getFixTime());
193         trip.setStartAddress(startTrip.getAddress());
194 
195         trip.setEndPositionId(endTrip.getId());
196         trip.setEndLat(endTrip.getLatitude());
197         trip.setEndLon(endTrip.getLongitude());
198         trip.setEndTime(endTrip.getFixTime());
199         trip.setEndAddress(endTrip.getAddress());
200 
201         trip.setDistance(calculateDistance(startTrip, endTrip, !ignoreOdometer));
202         trip.setDuration(tripDuration);
203         trip.setAverageSpeed(speedSum / (endIndex - startIndex));
204         trip.setMaxSpeed(speedMax);
205         trip.setSpentFuel(calculateFuel(startTrip, endTrip));
206 
207         trip.setDriverUniqueId(findDriver(startTrip, endTrip));
208         trip.setDriverName(findDriverName(trip.getDriverUniqueId()));
209 
210         return trip;
211     }

213     private static StopReport calculateStop(ArrayList<Position> positions, int startIndex, int endIndex) {
214         Position startStop = positions.get(startIndex);
215         Position endStop = positions.get(endIndex);
216 
217         StopReport stop = new StopReport();
218 
219         long deviceId = startStop.getDeviceId();
220         stop.setDeviceId(deviceId);
221         stop.setDeviceName(Context.getIdentityManager().getById(deviceId).getName());
222 
223         stop.setPositionId(startStop.getId());
224         stop.setLatitude(startStop.getLatitude());
225         stop.setLongitude(startStop.getLongitude());
226         stop.setStartTime(startStop.getFixTime());
227         stop.setAddress(startStop.getAddress());
228         stop.setEndTime(endStop.getFixTime());
229 
230         long stopDuration = endStop.getFixTime().getTime() - startStop.getFixTime().getTime();
231         stop.setDuration(stopDuration);
232         stop.setSpentFuel(calculateFuel(startStop, endStop));
233 
234         long engineHours = 0;
235         for (int i = startIndex + 1; i <= endIndex; i++) {
236             if (positions.get(i).getBoolean(Position.KEY_IGNITION)
237                     && positions.get(i - 1).getBoolean(Position.KEY_IGNITION)) {
238                 engineHours += positions.get(i).getFixTime().getTime() - positions.get(i - 1).getFixTime().getTime();
239             }
240         }
241         stop.setEngineHours(engineHours);
242 
243         return stop;
244 
245     }

247     private static <T extends BaseReport> T calculateTripOrStop(ArrayList<Position> positions, int startIndex,
248             int endIndex, boolean ignoreOdometer, Class<T> reportClass) {
249         if (reportClass.equals(TripReport.class)) {
250             return (T) calculateTrip(positions, startIndex, endIndex, ignoreOdometer);
251         } else {
252             return (T) calculateStop(positions, startIndex, endIndex);
253         }
254     }

256     private static boolean isMoving(ArrayList<Position> positions, int index, TripsConfig tripsConfig) {
257         if (tripsConfig.getMinimalNoDataDuration() > 0) {
258             boolean beforeGap = index < positions.size() - 1
259                     && positions.get(index + 1).getFixTime().getTime() - positions.get(index).getFixTime().getTime()
260                     >= tripsConfig.getMinimalNoDataDuration();
261             boolean afterGap = index > 0
262                     && positions.get(index).getFixTime().getTime() - positions.get(index - 1).getFixTime().getTime()
263                     >= tripsConfig.getMinimalNoDataDuration();
264             if (beforeGap || afterGap) {
265                 return false;
266             }
267         }
268         if (positions.get(index).getAttributes().containsKey(Position.KEY_MOTION)
269                 && positions.get(index).getAttributes().get(Position.KEY_MOTION) instanceof Boolean) {
270             return positions.get(index).getBoolean(Position.KEY_MOTION);
271         } else {
272             return positions.get(index).getSpeed() > tripsConfig.getSpeedThreshold();
273         }
274     }

276     public static <T extends BaseReport> Collection<T> detectTripsAndStops(Collection<Position> positionCollection,
277             TripsConfig tripsConfig, boolean ignoreOdometer, Class<T> reportClass) {
278         Collection<T> result = new ArrayList<>();
279 
280         ArrayList<Position> positions = new ArrayList<>(positionCollection);
281         if (positions != null && !positions.isEmpty()) {
282             boolean trips = reportClass.equals(TripReport.class);
283             MotionEventHandler  motionHandler = new MotionEventHandler(tripsConfig);
284             DeviceState deviceState = new DeviceState();
285             deviceState.setMotionState(isMoving(positions, 0, tripsConfig));
286             int startEventIndex = trips == deviceState.getMotionState() ? 0 : -1;
287             int startNoEventIndex = -1;
288             for (int i = 0; i < positions.size(); i++) {
289                 Map<Event, Position> event = motionHandler.updateMotionState(deviceState, positions.get(i),
290                         isMoving(positions, i, tripsConfig));
291                 if (startEventIndex == -1
292                         && (trips != deviceState.getMotionState() && deviceState.getMotionPosition() != null
293                         || trips == deviceState.getMotionState() && event != null)) {
294                     startEventIndex = i;
295                     startNoEventIndex = -1;
296                 } else if (trips != deviceState.getMotionState() && startEventIndex != -1
297                         && deviceState.getMotionPosition() == null && event == null) {
298                     startEventIndex = -1;
299                 }
300                 if (startNoEventIndex == -1
301                         && (trips == deviceState.getMotionState() && deviceState.getMotionPosition() != null
302                         || trips != deviceState.getMotionState() && event != null)) {
303                     startNoEventIndex = i;
304                 } else if (startNoEventIndex != -1 && deviceState.getMotionPosition() == null && event == null) {
305                     startNoEventIndex = -1;
306                 }
307                 if (startEventIndex != -1 && startNoEventIndex != -1 && event != null
308                         && trips != deviceState.getMotionState()) {
309                     result.add(calculateTripOrStop(positions, startEventIndex, startNoEventIndex,
310                             ignoreOdometer, reportClass));
311                     startEventIndex = -1;
312                 }
313             }
314             if (startEventIndex != -1 && (startNoEventIndex != -1 || !trips)) {
315                 result.add(calculateTripOrStop(positions, startEventIndex,
316                             startNoEventIndex != -1 ? startNoEventIndex : positions.size() - 1,
317                             ignoreOdometer, reportClass));
318             }
319         }
320         return result;
321     }
