17  package org.traccar.api;
18  
19  import java.sql.SQLException;
20  import java.util.Collection;
21  import java.util.HashSet;
22  import java.util.Set;
23  
24  import javax.ws.rs.GET;
25  import javax.ws.rs.QueryParam;
26  
27  import org.traccar.Context;
28  import org.traccar.database.ExtendedObjectManager;
29  import org.traccar.model.BaseModel;
30  
31  public class ExtendedObjectResource<T extends BaseModel> extends BaseObjectResource<T> {
32  
33      public ExtendedObjectResource(Class<T> baseClass) {
34          super(baseClass);
35      }
36  
37      @GET
38      public Collection<T> get(
39              @QueryParam("all") boolean all, @QueryParam("userId") long userId, @QueryParam("groupId") long groupId,
40              @QueryParam("deviceId") long deviceId, @QueryParam("refresh") boolean refresh) throws SQLException {
41  
42          ExtendedObjectManager<T> manager = (ExtendedObjectManager<T>) Context.getManager(getBaseClass());
43          if (refresh) {
44              manager.refreshItems();
45          }
46  
47          Set<Long> result = new HashSet<>(getSimpleManagerItems(manager, all, userId));
48  
49          if (groupId != 0) {
50              Context.getPermissionsManager().checkGroup(getUserId(), groupId);
51              result.retainAll(manager.getGroupItems(groupId));
52          }
53  
54          if (deviceId != 0) {
55              Context.getPermissionsManager().checkDevice(getUserId(), deviceId);
56              result.retainAll(manager.getDeviceItems(deviceId));
57          }
58          return manager.getItems(result);
59  
60      }
61  
62  }
