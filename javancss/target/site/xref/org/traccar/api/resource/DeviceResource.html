16  package org.traccar.api.resource;
17  
18  import org.traccar.Context;
19  import org.traccar.api.BaseObjectResource;
20  import org.traccar.database.DeviceManager;
21  import org.traccar.model.Device;
22  import org.traccar.model.DeviceTotalDistance;
23  
24  import javax.ws.rs.Consumes;
25  import javax.ws.rs.GET;
26  import javax.ws.rs.PUT;
27  import javax.ws.rs.Path;
28  import javax.ws.rs.Produces;
29  import javax.ws.rs.QueryParam;
30  import javax.ws.rs.core.MediaType;
31  import javax.ws.rs.core.Response;
32  
33  import java.sql.SQLException;
34  import java.util.Collection;
35  import java.util.HashSet;
36  import java.util.List;
37  import java.util.Set;
38  
39  @Path("devices")
40  @Produces(MediaType.APPLICATION_JSON)
41  @Consumes(MediaType.APPLICATION_JSON)
42  public class DeviceResource extends BaseObjectResource<Device> {
43  
44      public DeviceResource() {
45          super(Device.class);
46      }
47  
48      @GET
49      public Collection<Device> get(
50              @QueryParam("all") boolean all, @QueryParam("userId") long userId,
51              @QueryParam("uniqueId") List<String> uniqueIds,
52              @QueryParam("id") List<Long> deviceIds) throws SQLException {
53          DeviceManager deviceManager = Context.getDeviceManager();
54          Set<Long> result = null;
55          if (all) {
56              if (Context.getPermissionsManager().getUserAdmin(getUserId())) {
57                  result = deviceManager.getAllItems();
58              } else {
59                  Context.getPermissionsManager().checkManager(getUserId());
60                  result = deviceManager.getManagedItems(getUserId());
61              }
62          } else if (uniqueIds.isEmpty() && deviceIds.isEmpty()) {
63              if (userId == 0) {
64                  userId = getUserId();
65              }
66              Context.getPermissionsManager().checkUser(getUserId(), userId);
67              result = deviceManager.getUserItems(userId);
68          } else {
69              result = new HashSet<Long>();
70              for (String uniqueId : uniqueIds) {
71                  Device device = deviceManager.getByUniqueId(uniqueId);
72                  Context.getPermissionsManager().checkDevice(getUserId(), device.getId());
73                  result.add(device.getId());
74              }
75              for (Long deviceId : deviceIds) {
76                  Context.getPermissionsManager().checkDevice(getUserId(), deviceId);
77                  result.add(deviceId);
78              }
79          }
80          return deviceManager.getItems(result);
81      }
82  
83      @Path("{id}/distance")
84      @PUT
85      public Response updateTotalDistance(DeviceTotalDistance entity) throws SQLException {
86          Context.getPermissionsManager().checkAdmin(getUserId());
87          Context.getDeviceManager().resetTotalDistance(entity);
88          return Response.noContent().build();
89      }
90  
91  }
